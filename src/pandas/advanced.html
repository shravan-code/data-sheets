This section covers advanced Pandas topics including MultiIndex, time series analysis, string operations, and categorical data.

MultiIndex (Hierarchical Indexing)

MultiIndex (also known as Hierarchical Index) allows you to have multiple levels of indexing on an axis. This is useful for representing multi-dimensional data in a 2D structure.

Creating MultiIndex

import pandas as pd
import numpy as np

# From arrays
arrays = [
    ['Engineering', 'Engineering', 'Marketing', 'Marketing'],
    ['Alice', 'Bob', 'Carol', 'Dave']
]
idx = pd.MultiIndex.from_arrays(arrays, names=['dept', 'name'])

df = pd.DataFrame({
    'salary': [95000, 72000, 110000, 68000],
    'bonus': [9500, 7200, 11000, 6800]
}, index=idx)

# From tuples
idx2 = pd.MultiIndex.from_tuples([
    ('Engineering', 'Alice'),
    ('Engineering', 'Bob'),
    ('Marketing', 'Carol'),
    ('Marketing', 'Dave')
])

# From product (all combinations)
idx3 = pd.MultiIndex.from_product([
    ['2022', '2023'],                  # First level
    ['Q1', 'Q2', 'Q3', 'Q4']           # Second level
], names=['year', 'quarter'])

# From DataFrame
idx4 = pd.MultiIndex.from_frame(df[['dept', 'name']])

Creating DataFrame with MultiIndex

# Create MultiIndex DataFrame
df = pd.DataFrame({
    'salary': [95000, 72000, 110000, 68000],
    'bonus': [9500, 7200, 11000, 6800]
}, index=idx)

# Create with column MultiIndex
columns = pd.MultiIndex.from_product([
    ['Salary', 'Bonus'],
    ['2023', '2024']
])

df = pd.DataFrame(
    np.random.randint(50000, 120000, (4, 4)),
    index=['Alice', 'Bob', 'Carol', 'Dave'],
    columns=columns
)

Accessing MultiIndex Data

# Access outer level (all Engineering rows)
df.loc['Engineering']

# Access specific row
df.loc[('Engineering', 'Alice')]

# Access using xs (cross-section)
df.xs('Alice', level='name')
df.xs('Engineering', level='dept')

# Access using slicers
df.loc[('Engineering', 'Alice'):('Marketing', 'Carol')]

# With xs on columns
df.xs('2023', level=1, axis=1)

MultiIndex Operations

# Reset to regular columns
df.reset_index()

# Convert index levels to columns
df.reset_index(level=0)          # Reset first level only

# Stack (pivot inner level to columns)
df.stack()

# Unstack (pivot inner level to rows)
df.unstack()

# Swap levels
df.swaplevel()

# Sort by index
df.sort_index()
df.sort_index(level=[0, 1])       # Sort by both levels

# Drop a level
df.droplevel(0)                   # Drop first level
df.droplevel(1)                   # Drop second level

Set and Reset Index

# Set index from columns
df.set_index('name')             # Single column
df.set_index(['dept', 'name'])   # Multiple columns = MultiIndex

# Reset index (index becomes column)
df.reset_index()
df.reset_index(drop=True)        # Discard old index

Time Series Analysis

Pandas has powerful time series capabilities including date range generation, resampling, and rolling windows.

Creating Date Ranges

# Daily date range
dr = pd.date_range('2024-01-01', periods=365, freq='D')

# Date range with start and end
dr = pd.date_range('2024-01-01', '2024-12-31', freq='D')

# Business days only
dr = pd.date_range('2024-01-01', '2024-12-31', freq='B')

# Month starts
dr = pd.date_range('2024-01-01', periods=12, freq='MS')

# Month ends
dr = pd.date_range('2024-01-01', periods=12, freq='ME')

# Quarter starts
dr = pd.date_range('2024-01-01', periods=4, freq='QS')

# Weekly on Monday
dr = pd.date_range('2024-01-01', periods=52, freq='W-MON')

# Business date range
dr = pd.bdate_range('2024-01-01', '2024-03-31')

# Period range (for periods, not timestamps)
pr = pd.period_range('2024-01', periods=12, freq='M')

# Timedelta range
tr = pd.timedelta_range('0 days', '10 days', freq='6H')

Resampling

Resampling is time-based groupby - it converts time series from one frequency to another.

Downsampling (High to Low Frequency)

# Create daily data
dates = pd.date_range('2024-01-01', periods=365, freq='D')
prices = pd.Series(100 + np.cumsum(np.random.randn(365)), index=dates)

# Monthly average (downsample)
monthly_avg = prices.resample('ME').mean()

# Monthly last value
monthly_last = prices.resample('ME').last()

# Monthly first value
monthly_first = prices.resample('MS').first()

# Weekly sum
weekly_sum = prices.resample('W').sum()

# Quarterly OHLC (Open, High, Low, Close)
quarterly = prices.resample('QE').ohlc()

# Annual aggregation
annual = prices.resample('YE').agg(['min', 'max', 'mean', 'std'])

Upsampling (Low to High Frequency)

# Upsample daily to hourly
hourly = prices.resample('H').ffill()        # Forward fill
hourly = prices.resample('H').bfill()        # Backward fill
hourly = prices.resample('H').interpolate('linear')  # Linear interpolation

Resample with Custom Aggregation

# Named aggregation
monthly = prices.resample('ME').agg(
    avg_price=('price', 'mean'),
    volatility=('price', 'std'),
    monthly_high=('price', 'max'),
    monthly_low=('price', 'min')
)

# Multiple aggregations on DataFrame
df_ts = pd.DataFrame({
    'price': prices,
    'volume': np.random.randint(1000, 5000, 365)
}, index=dates)

df_ts.resample('W').agg({
    'price': 'mean',
    'volume': 'sum'
})

Rolling, Expanding, and EWM

Window functions compute statistics over a sliding window of data.

Rolling Window

# 7-day rolling mean
prices.rolling(7).mean()

# 7-day rolling standard deviation
prices.rolling(7).std()

# 7-day rolling min/max
prices.rolling(7).min()
prices.rolling(7).max()

# 7-day rolling sum
prices.rolling(7).sum()

# 7-day rolling median
prices.rolling(7).median()

# 7-day rolling quantile
prices.rolling(7).quantile(0.75)

# With minimum periods (need at least 3 valid values)
prices.rolling(7, min_periods=3).mean()

# Centered window
prices.rolling(7, center=True).mean()

# Rolling with custom function
prices.rolling(30).apply(
    lambda x: (x[-1] - x[0]) / x[0], 
    raw=True
)

Rolling Correlation and Covariance

# 30-day rolling correlation
prices.rolling(30).corr(other_series)

# 30-day rolling covariance
prices.rolling(30).cov(other_series)

Expanding Window

# Expanding mean (all-time cumulative)
prices.expanding().mean()

# Expanding with minimum periods
prices.expanding(min_periods=5).std()

# Expanding min/max
prices.expanding().min()
prices.expanding().max()

Exponentially Weighted Moving (EWM)

# EWMA with span
prices.ewm(span=12).mean()

# EWMA with alpha
prices.ewm(alpha=0.3).mean()

# EWMA with half-life
prices.ewm(halflife=6).mean()

# EWMA with center of mass
prices.ewm(com=5).var()

Time Series Shift and Difference

Shift and diff are essential for time series feature engineering.

# Shift values (move data forward in time)
prices.shift(1)                  # Lag (previous day)
prices.shift(7)                   # 1 week lag
prices.shift(-1)                  # Lead (next day)

# Shift within groups
df.groupby('id')['price'].shift(1)

# Difference from previous
prices.diff()                     # 1-period difference
prices.diff(2)                    # 2-period difference

# Percentage change
prices.pct_change()               # (current - previous) / previous
prices.pct_change(periods=7)      # Weekly percentage change

# Change frequency
prices.asfreq('D')               # Convert to daily frequency
prices.asfreq('B')               # Convert to business days

Shift for Time-Based Selection

# Select by specific time
df.at_time('09:30')              # All rows at 9:30 AM
df.between_time('09:30', '16:00')  # Rows between times

# First/last n rows per day
df.first('1D')                   # First day of data
df.last('7D')                    # Last 7 days

String Operations

Pandas provides vectorized string operations through the .str accessor.

Case Operations

s = pd.Series(['alice', 'BOB', 'Carol'])

s.str.upper()                    # ALICE, BOB, CAROL
s.str.lower()                    # alice, bob, carol
s.str.title()                    # Alice, Bob, Carol
s.str.capitalize()               # Alice, Bob, Carol (first char only)
s.str.swapcase()                  # ALICE, bob, cAROL

Contains and Match

s = pd.Series(['alice@example.com', 'bob@test.org', 'carol'])

# Contains substring
s.str.contains('example')         # True, False, False
s.str.contains('test', na=False) # Handle NaN

# Starts with / ends with
s.str.startswith('a')            # True, False, False
s.str.endswith('.com')           # True, False, False

# Regex match (from start)
s.str.match(r'^[a-z]+@')         # True, False, False

# Full match (entire string)
s.str.fullmatch('alice@example.com')  # True, False, False

Find and Position

s = pd.Series(['alice', 'bob', 'carol'])

s.str.find('a')                  # Position of 'a' (-1 if not found)
s.str.rfind('a')                 # Position from right
s.str.index('a')                 # Like find but raises error
s.str.rindex('a')                # From right

# Count occurrences
s.str.count('a')                 # Number of 'a' per string

Extraction

s = pd.Series(['alice123', 'bob456', 'carol789'])

# Single character
s.str[0]                         # First character
s.str[0:3]                       # Slice

# Extract with regex (capture groups)
s.str.extract(r'(\w+)(\d+)')      # Returns DataFrame with groups
s.str.extract(r'(\w+)(\d+)', expand=False)  # Returns Series of tuples

# Extract all matches
s.str.extractall(r'(\w+)')        # For strings with multiple matches

# Find all matches
s.str.findall(r'\d')             # List of matches

Replace

s = pd.Series(['alice', 'bob', 'carol'])

# Simple replace
s.str.replace('alice', 'Alice')  # Replace one string

# Regex replace
s.str.replace(r'\s+', '_', regex=True)  # Replace whitespace

# Replace with limit
s.str.replace('o', 'O', n=1)    # Replace first occurrence only

Split and Join

s = pd.Series(['alice@example.com', 'bob@test.com'])

# Split
s.str.split('@')                 # Returns list
s.str.split('@', expand=True)    # Returns DataFrame

# Split with limit
s.str.split('@', n=1)           # Max 1 split
s.str.rsplit('@', n=1)           # Split from right

# Join
pd.Series([['a', 'b'], ['c', 'd']]).str.join('-')  # 'a-b', 'c-d'

Padding and Alignment

s = pd.Series(['1', '12', '123'])

# Pad with zeros (zfill)
s.str.zfill(5)                  # '00001', '00012', '00123'

# Left/right pad
s.str.pad(5, side='left')       # '    1'
s.str.pad(5, side='right')      # '1    '

# Center
s.str.center(5, fillchar='-')   # '--1--'

Strip and Clean

s = pd.Series(['  alice  ', '\tbob\n', '  carol'])

s.str.strip()                    # Remove leading/trailing whitespace
s.str.lstrip()                   # Remove leading whitespace
s.str.rstrip()                   # Remove trailing whitespace

# Remove specific characters
s.str.strip(' \t\n')             # Remove whitespace chars

String Length and Empty Check

s = pd.Series(['alice', 'bob', ''])

s.str.len()                     # Length of each string
s.str.len().sum()               # Total length

# Check if empty
s.str.len() == 0
s.str.isalpha()                 # All alphabetic
s.str.isdigit()                 # All digits
s.str.isalnum()                 # Alphanumeric
s.str.isspace()                 # All whitespace
s.str.islower()                 # All lowercase
s.str.isupper()                 # All uppercase

Categorical Data

Categorical data is used for columns with a limited number of unique values. It saves memory and enables better performance.

Creating Categorical Data

# Create categorical directly
cat = pd.Categorical(['a', 'b', 'a', 'c', 'b'])

# Ordered categorical
cat = pd.Categorical(
    ['a', 'b', 'a', 'c'],
    categories=['c', 'b', 'a'],
    ordered=True
)

# Convert column to category
df['dept'] = df['dept'].astype('category')

# With category dtype
df['grade'] = pd.Categorical(
    ['A', 'B', 'A', 'C'],
    categories=['C', 'B', 'A'],
    ordered=True
)

Category Accessor Methods

# Access categories
df['dept'].cat.categories        # Index(['Engineering', 'Marketing', 'HR'])

# Access integer codes
df['dept'].cat.codes             # 0, 1, 2...

# Check if ordered
df['dept'].cat.ordered          # True or False

Modify Categories

# Rename categories
df['dept'].cat.rename_categories({'Engineering': 'Eng', 
                                   'Marketing': 'Mkt'})

# Reorder categories
df['dept'].cat.reorder_categories(['HR', 'Marketing', 'Engineering'])

# Add categories
df['dept'].cat.add_categories(['Legal'])

# Remove categories
df['dept'].cat.remove_categories(['Legal'])

# Remove unused categories
df['dept'].cat.remove_unused_categories()

# Set categories (replace all)
df['dept'].cat.set_categories(['Engineering', 'Marketing'])

Categorical Operations

# Make ordered
df['dept'].cat.as_ordered()
df['dept'].cat.as_unordered()

# Comparisons (for ordered categories)
df['grade'] = pd.Categorical(
    ['A', 'B', 'A', 'C'],
    categories=['C', 'B', 'A'],
    ordered=True
)

df[df['grade'] >= 'B']           # Works for ordered categories

DataFrame Operations

Advanced DataFrame operations for data manipulation.

Eval and Query

# Use eval for efficient column operations
df.eval('salary + bonus')

# Assign with eval
df.eval('total = salary + bonus', inplace=True)

# Query with boolean expressions
df.query('salary > 80000 and dept == "Engineering"')

# Use local variables
threshold = 80000
df.query('salary > @threshold')

Where and Mask

# Where (replace where condition is False)
df['salary'].where(df['salary'] > 0, other=0)

# Where with DataFrame condition
df.where(df > 0)

# Mask (replace where condition is True)
df['salary'].mask(df['salary'] > 100000, other=100000)

Boolean Operations

# All/Any
df.all()                        # All values True
df.any()                        # Any value True

# Element-wise comparison
df['salary'].gt(80000)          # Greater than
df['salary'].lt(100000)         # Less than
df['salary'].ge(80000)          # Greater or equal
df['salary'].le(100000)         # Less or equal
df['salary'].eq(95000)           # Equal
df['salary'].ne(95000)           # Not equal

# Between
df['salary'].between(80000, 100000)

# Is in
df['dept'].isin(['Engineering', 'HR'])

Clip and Round

# Clip values
df['salary'].clip(lower=50000, upper=150000)

# Round values
df['salary'].round(-3)          # Round to nearest 1000
df['salary'].round(2)           # Round to 2 decimals

# Floor and ceil
df['salary'].floor(1000)        # Round down to 1000
df['salary'].ceil(1000)         # Round up to 1000

Abs and Sign

# Absolute values
df['change'].abs()

# Sign
df['change'].sign()             # -1, 0, or 1

Sample and Random Operations

# Random sample
df.sample(n=10)                 # 10 random rows
df.sample(frac=0.1)            # 10% of rows
df.sample(frac=0.5, replace=True)  # With replacement

# Set random seed
df.sample(n=10, random_state=42)
