Pipelines

This section covers building ETL (Extract, Transform, Load) pipelines using Python and Pandas, including patterns for batch processing, data validation, and workflow orchestration.

What is an ETL Pipeline?

ETL stands for Extract, Transform, Load. It is a data engineering pattern that processes data in discrete, scheduled chunks rather than continuously.

The Three Phases

  • Extract: Read raw data from sources (databases, files, APIs)
  • Transform: Clean, validate, enrich, and reshape the data
  • Load: Write processed data to destination

Batch vs Streaming

AspectBatch ETLStreaming
LatencyMinutes to hoursMilliseconds
Data VolumeVery largeContinuous
Use CaseReports, BI dashboardsReal-time analytics

import pandas as pd
import os
import logging

logger = logging.getLogger(__name__)

EXPECTED_COLUMNS = {'order_id', 'order_date', 'product_id', 'quantity', 'unit_price', 'customer_id'}

def extract_from_csv(filepath: str) -> pd.DataFrame:
    """Extract data from CSV file with validation."""
    
    if not os.path.exists(filepath):
        raise FileNotFoundError(f"Source file not found: {filepath}")
    
    df = pd.read_csv(filepath)
    
    # Validate columns
    missing_cols = EXPECTED_COLUMNS - set(df.columns)
    if missing_cols:
        raise ValueError(f"Missing columns: {missing_cols}")
    
    logger.info(f"Extracted {len(df)} rows from {filepath}")
    return df

def extract_from_csv_chunks(filepath: str, chunksize: int = 10000) -> pd.DataFrame:
    """Extract large CSV files in chunks."""
    chunks = []
    for chunk in pd.read_csv(filepath, chunksize=chunksize):
        chunks.append(chunk)
    return pd.concat(chunks, ignore_index=True)

Extract from API

import requests
import pandas as pd
import time
import logging

logger = logging.getLogger(__name__)

BASE_URL = 'https://api.example.com/v1'
API_KEY = 'your_api_key'

def extract_from_api(endpoint: str, params: dict = None) -> pd.DataFrame:
    """Extract data from REST API with pagination."""
    
    headers = {'Authorization': f'Bearer {API_KEY}'}
    all_records = []
    page = 1
    
    while True:
        params = params or {}
        params['page'] = page
        params['per_page'] = 100
        
        response = requests.get(
            f"{BASE_URL}/{endpoint}",
            headers=headers,
            params=params,
            timeout=30
        )
        response.raise_for_status()
        
        data = response.json()
        records = data.get('data', [])
        
        if not records:
            break
        
        all_records.extend(records)
        logger.info(f"Page {page}: {len(records)} records")
        
        if not data.get('has_more', False):
            break
        
        page += 1
        time.sleep(0.2)  # Rate limiting
    
    logger.info(f"Total extracted: {len(all_records)} records")
    return pd.DataFrame(all_records)

def extract_from_s3(bucket: str, key: str) -> pd.DataFrame:
    """Extract data from S3."""
    
    s3 = boto3.client('s3')
    
    csv_obj = s3.get_object(Bucket=bucket, Key=key)
    body = csv_obj['Body']
    csv_string = body.read().decode('utf-8')
    
    df = pd.read_csv(StringIO(csv_string))
    return df

def extract_parquet_from_s3(bucket: str, prefix: str) -> pd.DataFrame:
    """Extract partitioned Parquet files from S3."""
    
    s3 = boto3.resource('s3')
    bucket_obj = s3.Bucket(bucket)
    
    dfs = []
    for obj in bucket_obj.objects.filter(Prefix=prefix):
        if obj.key.endswith('.parquet'):
            df = pd.read_parquet(f's3://{bucket}/{obj.key}')
            dfs.append(df)
    
    if dfs:
        return pd.concat(dfs, ignore_index=True)
    return pd.DataFrame()

Basic Transformations

import pandas as pd
import numpy as np
from datetime import datetime

def transform_sales_data(df: pd.DataFrame) -> pd.DataFrame:
    """Transform sales data with cleaning and enrichment."""
    
    # 1. Strip whitespace from string columns
    for col in df.select_dtypes(include='object').columns:
        df[col] = df[col].str.strip() if df[col].dtype == 'object' else df[col]
    
    # 2. Type casting
    df['quantity'] = pd.to_numeric(df['quantity'], errors='coerce')
    df['unit_price'] = pd.to_numeric(df['unit_price'], errors='coerce')
    df['order_date'] = pd.to_datetime(df['order_date'], errors='coerce')
    
    # 3. Remove rows with null critical fields
    df = df.dropna(subset=['order_id', 'quantity', 'unit_price'])
    
    # 4. Remove duplicates
    df = df.drop_duplicates(subset=['order_id'])
    
    # 5. Add derived fields
    df['revenue'] = df['quantity'] * df['unit_price']
    df['etl_run_date'] = datetime.now().isoformat()
    
    # 6. Add time components
    df['year'] = df['order_date'].dt.year
    df['month'] = df['order_date'].dt.month
    df['day'] = df['order_date'].dt.day
    
    return df

def clean_nulls(df: pd.DataFrame, strategy: dict = None) -> pd.DataFrame:
    """Handle missing values based on strategy."""
    
    strategy = strategy or {}
    
    for col, method in strategy.items():
        if method == 'drop':
            df = df.dropna(subset=[col])
        elif method == 'mean':
            df[col] = df[col].fillna(df[col].mean())
        elif method == 'median':
            df[col] = df[col].fillna(df[col].median())
        elif method == 'mode':
            df[col] = df[col].fillna(df[col].mode()[0])
        elif method == 'forward':
            df[col] = df[col].ffill()
        elif method == 'backward':
            df[col] = df[col].bfill()
        elif method == 'zero':
            df[col] = df[col].fillna(0)
        elif method == 'empty':
            df[col] = df[col].fillna('')
    
    return df

Data Validation

def validate_data(df: pd.DataFrame, schema: dict) -> tuple[bool, list]:
    """Validate data against schema rules.
    
    Args:
        df: DataFrame to validate
        schema: Dict with validation rules
    
    Returns:
        (is_valid, errors): Tuple of validity and list of errors
    """
    
    errors = []
    
    for col, rules in schema.items():
        # Check column exists
        if col not in df.columns:
            errors.append(f"Missing column: {col}")
            continue
        
        # Check for nulls
        if rules.get('not_null', False):
            null_count = df[col].isna().sum()
            if null_count > 0:
                errors.append(f"{col}: {null_count} null values found")
        
        # Check data type
        if 'dtype' in rules:
            expected_dtype = rules['dtype']
            if df[col].dtype != expected_dtype:
                errors.append(f"{col}: Expected {expected_dtype}, got {df[col].dtype}")
        
        # Check min/max
        if 'min' in rules:
            below_min = (df[col] < rules['min']).sum()
            if below_min > 0:
                errors.append(f"{col}: {below_min} values below minimum {rules['min']}")
        
        if 'max' in rules:
            above_max = (df[col] > rules['max']).sum()
            if above_max > 0:
                errors.append(f"{col}: {above_max} values above maximum {rules['max']}")
        
        # Check allowed values
        if 'allowed_values' in rules:
            invalid = ~df[col].isin(rules['allowed_values'])
            if invalid.sum() > 0:
                errors.append(f"{col}: Invalid values found")
    
    is_valid = len(errors) == 0
    return is_valid, errors

Enrichment

def enrich_with_lookup(df: pd.DataFrame, lookup_df: pd.DataFrame, 
                       key: str, columns: list) -> pd.DataFrame:
    """Enrich data by joining with lookup table."""
    
    lookup = lookup_df[[key] + columns].drop_duplicates(subset=[key])
    return df.merge(lookup, on=key, how='left')

def add_business_logic(df: pd.DataFrame) -> pd.DataFrame:
    """Add business logic calculations."""
    
    # Customer tier based on lifetime value
    customer_value = df.groupby('customer_id')['revenue'].sum()
    df['customer_tier'] = pd.cut(
        df['customer_id'].map(customer_value),
        bins=[0, 500, 2000, 10000, np.inf],
        labels=['Bronze', 'Silver', 'Gold', 'Platinum']
    )
    
    # Discount category
    df['discount_category'] = pd.cut(
        df['discount_pct'],
        bins=[-np.inf, 0, 10, 20, np.inf],
        labels=['None', 'Low', 'Medium', 'High']
    )
    
    return df

def aggregate_to_summary(df: pd.DataFrame) -> pd.DataFrame:
    """Create summary aggregations."""
    
    summary = df.groupby(['customer_id', 'order_date']).agg(
        order_count=('order_id', 'count'),
        total_revenue=('revenue', 'sum'),
        avg_discount=('discount_pct', 'mean'),
        item_count=('quantity', 'sum')
    ).reset_index()
    
    return summary

UPSERT_SQL = '''
    INSERT INTO sales_fact 
        (order_id, order_date, product_id, customer_id, 
         quantity, unit_price, revenue, etl_run_date)
    VALUES %s
    ON CONFLICT (order_id) DO UPDATE SET
        quantity = EXCLUDED.quantity,
        unit_price = EXCLUDED.unit_price,
        revenue = EXCLUDED.revenue,
        etl_run_date = EXCLUDED.etl_run_date;
'''

def upsert_to_sql(df: pd.DataFrame, table_name: str) -> int:
    """Upsert data to SQL table (insert or update on conflict)."""
    
    if df.empty:
        return 0
    
    engine = create_engine(DB_CONFIG)
    
    columns = ['order_id', 'order_date', 'product_id', 'customer_id',
               'quantity', 'unit_price', 'revenue', 'etl_run_date']
    
    values = [tuple(row[col] for col in columns) for _, row in df.iterrows()]
    
    with engine.begin() as conn:
        psycopg2.extras.execute_values(
            conn, 
            UPSERT_SQL, 
            values,
            page_size=500
        )
    
    engine.dispose()
    return len(df)

def load_to_staging_table(df: pd.DataFrame, table_name: str) -> int:
    """Load to staging table, then merge to target table."""
    
    engine = create_engine(DB_CONFIG)
    stage_table = f"{table_name}_stage"
    
    # Write to staging table
    df.to_sql(stage_table, engine, if_exists='replace', index=False)
    
    # Merge from staging to target
    merge_sql = f'''
        DELETE FROM {table_name} 
        WHERE order_date IN (SELECT order_date FROM {stage_table});
        
        INSERT INTO {table_name} 
        SELECT * FROM {stage_table};
        
        DROP TABLE {stage_table};
    '''
    
    with engine.begin() as conn:
        conn.execute(text(merge_sql))
    
    engine.dispose()
    return len(df)

Load to S3

import pandas as pd
import boto3
from io import BytesIO

def load_to_s3(df: pd.DataFrame, bucket: str, key: str, format: str = 'csv') -> None:
    """Load DataFrame to S3."""
    
    s3 = boto3.client('s3')
    
    if format == 'csv':
        buffer = BytesIO()
        df.to_csv(buffer, index=False)
        content_type = 'text/csv'
    elif format == 'parquet':
        buffer = BytesIO()
        df.to_parquet(buffer, index=False, engine='pyarrow')
        content_type = 'application/octet-stream'
    elif format == 'json':
        buffer = BytesIO()
        df.to_json(buffer, orient='records', lines=True)
        content_type = 'application/json'
    
    buffer.seek(0)
    s3.put_object(Bucket=bucket, Key=key, Body=buffer.getvalue(),
                    ContentType=content_type)

def load_partitioned_parquet(df: pd.DataFrame, bucket: str, base_path: str) -> None:
    """Load Parquet files partitioned by year/month."""
    
    df['year'] = df['order_date'].dt.year
    df['month'] = df['order_date'].dt.month
    
    for (year, month), partition_df in df.groupby(['year', 'month']):
        key = f"{base_path}/year={year}/month={month:02d}/data.parquet"
        buffer = BytesIO()
        partition_df.drop(columns=['year', 'month']).to_parquet(
            buffer, index=False, engine='pyarrow', compression='snappy'
        )
        buffer.seek(0)
        
        s3 = boto3.client('s3')
        s3.put_object(Bucket=bucket, Key=key, Body=buffer.getvalue())

Batch Processing Pipeline

import pandas as pd
import os
from datetime import datetime, timedelta

def process_date_range(start_date: str, end_date: str) -> pd.DataFrame:
    """Process data for a range of dates."""
    
    all_data = []
    current_date = pd.to_datetime(start_date)
    end = pd.to_datetime(end_date)
    
    while current_date <= end:
        date_str = current_date.strftime('%Y%m%d')
        
        filepath = f"{RAW_PATH}/sales_{date_str}.csv"
        
        if os.path.exists(filepath):
            df = pd.read_csv(filepath)
            df['source_date'] = current_date
            all_data.append(df)
            print(f"Processed {date_str}: {len(df)} rows")
        else:
            print(f"File not found: {filepath}")
        
        current_date += timedelta(days=1)
    
    if all_data:
        combined = pd.concat(all_data, ignore_index=True)
        return combined
    
    return pd.DataFrame()

def incremental_process(last_run_date: str, current_date: str) -> pd.DataFrame:
    """Process only new data since last run (incremental load)."""
    
    all_records = []
    
    # Read all files in date range
    current = pd.to_datetime(last_run_date)
    end = pd.to_datetime(current_date)
    
    while current <= end:
        date_str = current.strftime('%Y%m%d')
        filepath = f"{RAW_PATH}/sales_{date_str}.csv"
        
        if os.path.exists(filepath):
            df = pd.read_csv(filepath)
            df['etl_run_date'] = datetime.now()
            all_records.append(df)
        
        current += timedelta(days=1)
    
    return pd.concat(all_records, ignore_index=True) if all_records else pd.DataFrame()

Error Handling and Retry

import time
from functools import wraps

def retry(max_attempts: int = 3, delay: int = 1):
    """Decorator for retrying failed operations."""
    
    def decorator(func):
        @wraps(func)
        def wrapper(*args, **kwargs):
            attempts = 0
            while attempts < max_attempts:
                try:
                    return func(*args, **kwargs)
                except Exception as e:
                    attempts += 1
                    if attempts >= max_attempts:
                        raise
                    logger.warning(f"Attempt {attempts} failed: {e}. Retrying...")
                    time.sleep(delay * attempts)
            return None
        return wrapper
    return decorator

@retry(max_attempts=3, delay=2)
def load_with_retry(df: pd.DataFrame, table_name: str) -> int:
    """Load with automatic retry on failure."""
    return load_to_sql(df, table_name)

Best Practices

Idempotency

Running the same pipeline twice should produce the same result. This is essential for safe reruns.

# Use upsert instead of insert
# Always include etl_run_date for tracking
# Use WHERE clauses to delete before insert
# Include checksum/hash for deduplication

def idempotent_load(df: pd.DataFrame, table_name: str) -> int:
    """Load data idempotently."""
    
    # Add run identifier
    run_id = datetime.now().strftime('%Y%m%d_%H%M%S')
    df['etl_run_id'] = run_id
    
    # Delete existing records for the same date
    load_date = df['order_date'].iloc[0]
    
    engine = create_engine(DB_CONFIG)
    with engine.begin() as conn:
        conn.execute(
            text(f"DELETE FROM {table_name} WHERE order_date = '{load_date}'")
        )
        df.to_sql(table_name, conn, if_exists='append', index=False)
    
    engine.dispose()
    return len(df)

Extract-Transform-Load (ETL)

def etl_pipeline(source: str, destination: str, 
                  transform_fn) -> int:
    """Full ETL pipeline."""
    
    # Extract
    df = pd.read_csv(source)
    
    # Transform
    df = transform_fn(df)
    
    # Load
    df.to_parquet(destination)

def pipeline_with_staging(df: pd.DataFrame, table: str) -> int:
    """Pipeline that uses staging table."""
    
    # 1. Write to staging
    stage_table = f"{table}_stage"
    df.to_sql(stage_table, engine, if_exists='replace', index=False)
    
    # 2. Validate staging data
    stage_df = pd.read_sql(f"SELECT * FROM {stage_table}", engine)
    
    if stage_df.empty:
        logger.warning("Staging table is empty")
        return 0
    
    # 3. Upsert to target table
    merge_sql = f"""
        INSERT INTO {table} 
        SELECT * FROM {stage_table}
        ON CONFLICT (id) DO UPDATE SET
            col1 = EXCLUDED.col1,
            col2 = EXCLUDED.col2,
            updated_at = NOW();
    """
    
    with engine.begin() as conn:
        conn.execute(text(merge_sql))
        conn.execute(text(f"DROP TABLE {stage_table}"))
    
    return len(df)
