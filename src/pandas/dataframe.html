DataFrame

Complete reference of all DataFrame methods organized by category.

Creation Methods

pd.DataFrame() - Main Constructor

# From dictionary of lists
df = pd.DataFrame({
    'name': ['Alice', 'Bob', 'Carol'],
    'age': [34, 28, 45],
    'salary': [95000, 72000, 110000]
})

# From list of dictionaries
df = pd.DataFrame([
    {'name': 'Alice', 'age': 34, 'salary': 95000},
    {'name': 'Bob', 'age': 28, 'salary': 72000}
])

# From 2D array
import numpy as np
arr = np.array([[1, 2, 3], [4, 5, 6]])
df = pd.DataFrame(arr, columns=['A', 'B', 'C'])

# With index
df = pd.DataFrame(data, index=['row1', 'row2', 'row3'])

# With dtypes
df = pd.DataFrame(data, dtype='float32')

Other Creation Methods

# From dict of arrays
pd.DataFrame.from_dict({'a': [1, 2], 'b': [3, 4]})

# From records (structured array)
pd.DataFrame.from_records([('Alice', 34), ('Bob', 28)])

# From other DataFrame
pd.DataFrame(df2)

# Empty DataFrame
df = pd.DataFrame()

Attributes

Attributes provide information about the DataFrame structure.

df.shape               # (rows, cols) tuple
df.columns             # Column labels (Index)
df.index               # Row labels (Index)
df.dtypes              # Series with dtype per column
df.ndim                # Number of dimensions (always 2)
df.size                # Total elements (rows * cols)
df.empty               # True if DataFrame is empty
df.axes                # [row_index, column_index]
df.T                   # Transpose (swap rows and columns)
df.name                # Name of DataFrame (None by default)
df.memory_usage(deep=True)  # Memory usage in bytes

Accessing Attributes

# Get column names
df.columns.tolist()    # List of column names

# Get index as list
df.index.tolist()      # List of row labels

# Check if column exists
'column_name' in df.columns

# Get dtypes as dict
df.dtypes.to_dict()

Indexing and Selection Methods

Selection Indexers

# Label-based selection
df.loc[row_label, col_label]
df.loc[row_slice, col_slice]
df.loc[bool_mask]
df.loc[condition]

# Position-based selection
df.iloc[row_pos, col_pos]
df.iloc[row_slice, col_slice]
df.iloc[bool_mask]

# Single value (fastest)
df.at[row_label, col_label]   # Label
df.iat[row_pos, col_pos]       # Position

# Column selection shorthand
df['column']                   # Single column (Series)
df[['col1', 'col2']]           # Multiple columns (DataFrame)
df.column                      # Attribute-style (if valid identifier)

Boolean Selection

# Filter rows
df[df['column'] > value]
df[df['column'].isin(['a', 'b'])]
df[df['column'].between(lower, upper)]

# Query method
df.query('column > value')
df.query('column > @variable')  # Using local variable

Filter Methods

# Filter columns by name pattern
df.filter(like='salary')       # Columns containing 'salary'
df.filter(regex='^Q[1-4]_')     # Columns matching regex

# Filter by items
df.filter(items=['col1', 'col2'])

# Select by dtype
df.select_dtypes(include=['number'])
df.select_dtypes(include='object')
df.select_dtypes(exclude=['datetime'])

Head, Tail, Sample

df.head()              # First 5 rows
df.head(10)            # First 10 rows
df.tail()              # Last 5 rows
df.tail(3)             # Last 3 rows
df.sample()            # 1 random row
df.sample(n=5)         # 5 random rows
df.sample(frac=0.1)    # 10% of rows
df.sample(n=5, random_state=42)  # Reproducible

Data Manipulation Methods

Add/Modify Columns

# Add column
df['new_col'] = values
df['bonus'] = df['salary'] * 0.1

# Assign (returns new DataFrame)
df = df.assign(bonus=df['salary'] * 0.1)
df = df.assign(
    bonus=lambda x: x['salary'] * 0.1,
    total=lambda x: x['salary'] + x['bonus']
)

# Insert at position
df.insert(loc=2, column='emp_id', value=range(1, len(df)+1))

# Pop (remove and return)
col = df.pop('column_name')

Remove Columns/Rows

# Drop columns
df.drop(columns=['col1', 'col2'])
df.drop('column', axis=1)
df.drop(['col1', 'col2'], axis=1)

# Drop rows
df.drop(index=['row1', 'row2'])
df.drop([0, 1, 2])              # By position

# Drop duplicates
df.drop_duplicates()                       # Keep first
df.drop_duplicates(subset=['col'])          # On column(s)
df.drop_duplicates(keep='last')             # Keep last

Rename

# Rename columns
df.rename(columns={'old': 'new', 'old2': 'new2'})
df.rename(columns=str.upper)      # Function
df.columns = ['a', 'b', 'c']      # Direct assignment

# Rename index
df.rename(index={'row1': 'new_row1'})
df.rename(index=str.lower)

# Rename axis
df.rename_axis('rows').rename_axis('cols', axis=1)

Add Prefix/Suffix

df.add_prefix('col_')      # Prefix to columns
df.add_suffix('_data')     # Suffix to columns
df.add_suffix('_data').add_prefix('row_')  # Both

Sorting Methods

Sort by Values

# Single column ascending
df.sort_values('salary')

# Single column descending
df.sort_values('salary', ascending=False)

# Multiple columns
df.sort_values(['dept', 'salary'], ascending=[True, False])

# With NaN position
df.sort_values('salary', na_position='last')  # 'first' or 'last'

# With custom key
df.sort_values('name', key=lambda x: x.str.lower())
df.sort_values('salary', key=lambda x: abs(x))

Sort by Index

# Sort by row index
df.sort_index()

# Sort by column index
df.sort_index(axis=1)

# Descending
df.sort_index(ascending=False)

Rank and Top/Bottom

# Rank values
df['rank'] = df['salary'].rank()
df['rank'] = df['salary'].rank(method='dense')  # No gaps
df['rank'] = df['salary'].rank(method='first')   # By order
df['rank'] = df['salary'].rank(ascending=False)  # 1 = highest

# Top/Bottom N
df.nlargest(5, 'salary')       # Top 5 by salary
df.nsmallest(3, 'age')         # Bottom 3 by age
df.nlargest(5, ['salary', 'age'])  # Multi-column

Missing Data Methods

Detect Missing Values

df.isna()            # Returns True for NaN/None
df.isnull()          # Alias for isna()
df.notna()           # Returns True for non-NaN
df.notnull()         # Alias for notna()

# Count missing per column
df.isna().sum()

# Count missing per row
df.isna().sum(axis=1)

# Check if any/all missing
df.isna().any()       # Per column
df.isna().all()       # Per column

Fill Missing Values

# Fill with value
df.fillna(0)
df.fillna({'col1': 0, 'col2': 'unknown'})

# Forward fill (last valid value)
df.fillna(method='ffill')
df['col'].ffill()

# Backward fill (next valid value)
df.fillna(method='bfill')
df['col'].bfill()

# Fill with mean/median
df['col'].fillna(df['col'].mean())
df['col'].fillna(df['col'].median())

# Fill with interpolation
df.interpolate()
df.interpolate(method='linear')
df.interpolate(method='polynomial', order=2)

Drop Missing Values

# Drop rows with any missing
df.dropna()

# Drop rows with all missing
df.dropna(how='all')

# Drop rows missing in specific columns
df.dropna(subset=['col1', 'col2'])

# Drop columns with missing
df.dropna(axis=1)

# Keep rows with at least N non-null values
df.dropna(thresh=5)

Replace Values

# Replace specific values
df.replace(old_value, new_value)
df.replace([old1, old2], new_value)

# Replace with dict
df.replace({'col1': old1, 'col2': old2}, new_value)

# Dict-based replacement
df.replace({'col': {old1: new1, old2: new2}})

# Regex replacement
df.replace(r'\s+', '', regex=True)

Where and Mask

# Where (replace where False)
df.where(df > 0, other=0)

# Mask (replace where True)
df.mask(df > 100, other=100)

Aggregation Methods

Basic Aggregations

# Per column aggregations
df.mean()             # Mean of each numeric column
df.sum()              # Sum of each column
df.std()              # Standard deviation
df.var()              # Variance
df.min()              # Minimum
df.max()              # Maximum
df.count()            # Count non-null values
df.median()           # Median
df.mode()             # Mode (most frequent)
df.prod()             # Product of values
df.abs()              # Absolute values

# Along axis
df.mean(axis=0)       # Per column (default)
df.mean(axis=1)       # Per row

Describe

df.describe()                   # Numeric columns only
df.describe(include='all')       # All columns
df.describe(include='object')    # Object columns only
df.describe(percentiles=[.1, .5, .9])  # Custom percentiles

agg() - Multiple Aggregations

# Apply multiple functions
df.agg(['mean', 'std', 'min', 'max'])

# Different functions per column
df.agg({
    'col1': ['mean', 'sum'],
    'col2': ['min', 'max']
})

# Named aggregation (recommended)
df.agg(
    mean_col1=('col1', 'mean'),
    sum_col2=('col2', 'sum')
)

aggregate()

# Same as agg
df.aggregate(['mean', 'sum'])

apply() - Custom Functions

# Apply function to each row/column
df.apply(np.sum)              # Sum per column
df.apply(np.sum, axis=1)     # Sum per row

# Custom function
df.apply(lambda x: x.max() - x.min())

# Element-wise (deprecated, use map)
df.applymap(lambda x: x ** 2)

transform()

# Transform returns same shape
df.transform(np.abs)                 # Absolute values
df.transform([np.abs, np.sqrt])       # Multiple functions
df.transform(lambda x: x / x.sum())   # Custom

GroupBy Methods

Creating GroupBy

# Group by column
g = df.groupby('dept')

# Group by multiple columns
g = df.groupby(['dept', 'active'])

# Group by index
g = df.groupby(level=0)
g = df.groupby(level=['dept', 'name'])

GroupBy Aggregation

# Aggregate single column
df.groupby('dept')['salary'].mean()
df.groupby('dept')['salary'].agg('mean')

# Aggregate entire DataFrame
df.groupby('dept').mean()
df.groupby('dept').agg(['mean', 'max'])

GroupBy Transform

# Add group mean to each row
df['dept_avg'] = df.groupby('dept')['salary'].transform('mean')

# Z-score within group
df['z_score'] = df.groupby('dept')['salary'].transform(
    lambda x: (x - x.mean()) / x.std()
)

# Rank within group
df['rank'] = df.groupby('dept')['salary'].transform('rank')

GroupBy Filter

# Keep groups with condition
df.groupby('dept').filter(lambda x: len(x) > 2)
df.groupby('dept').filter(lambda x: x['salary'].mean() > 80000)

GroupBy Apply

# Apply custom function to each group
def normalize(group):
    return (group - group.mean()) / group.std()

df.groupby('dept')['salary'].apply(normalize)

Cumulative Methods

# Cumulative within group
df.groupby('dept')['salary'].cumsum()
df.groupby('dept')['salary'].cumprod()
df.groupby('dept')['salary'].cummax()
df.groupby('dept')['salary'].cummin()
df.groupby('dept')['salary'].cumcount()  # 0, 1, 2, ...

Rolling/Expanding within Group

# Rolling within group
df.groupby('dept')['salary'].rolling(3).mean()

# Expanding within group
df.groupby('dept')['salary'].expanding().mean()

Merge and Join Methods

pd.merge()

# Basic merge
pd.merge(df1, df2, on='key')

# Different column names
pd.merge(df1, df2, left_on='key1', right_on='key2')

# Join types
pd.merge(df1, df2, on='key', how='inner')    # Only matching
pd.merge(df1, df2, on='key', how='left')     # All left
pd.merge(df1, df2, on='key', how='right')    # All right
pd.merge(df1, df2, on='key', how='outer')    # All

# Cross join
pd.merge(df1, df2, how='cross')

# With suffixes
pd.merge(df1, df2, on='key', suffixes=('_df1', '_df2'))

# Validation
pd.merge(df1, df2, on='key', validate='one_to_one')

# With indicator
pd.merge(df1, df2, on='key', indicator=True)  # Shows merge source

df.join()

# Join on index
df1.join(df2)                    # Left join by default
df1.join(df2, how='inner')       # Inner
df1.join(df2, how='outer')       # Outer
df1.join(df2, how='right')      # Right

# With suffixes
df1.join(df2, lsuffix='_left', rsuffix='_right')

# Join with column as index
df1.join(df2.set_index('key'))

pd.concat()

# Concatenate rows
pd.concat([df1, df2])

# With ignore_index
pd.concat([df1, df2], ignore_index=True)

# With keys
pd.concat([df1, df2], keys=['2023', '2024'])

# Concatenate columns
pd.concat([df1, df2], axis=1)

# Inner join on columns
pd.concat([df1, df2], axis=1, join='inner')

Combine Methods

# Fill from other DataFrame
df1.combine_first(df2)          # Use df2 where df1 has NaN

# Update in place
df1.update(df2)                 # Update with non-NaN values from df2

# Element-wise combine
df1.combine(df2, func=np.maximum)

# Compare
df1.compare(df2)                # Show differences

Reshaping Methods

Pivot

    index='dept',
    columns='year',

# Melt multiple columns
pd.melt(df, id_vars=['name'], value_vars=['2023', '2024'])

Stack and Unstack

Wide to Long

Crosstab

Explode

Squeeze
