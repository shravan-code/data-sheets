Pandas Intermediate

Grouping, merging, pivoting, and data transformations.

Group By

# Group and aggregate
df.groupby('city')['age'].mean()
df.groupby('city').agg({'age': 'mean', 'name': 'count'})

# Multiple aggregations
df.groupby('city').agg(
    age_mean=('age', 'mean'),
    age_max=('age', 'max'),
    count=('name', 'count')
)

# Transform
df['age_rank'] = df.groupby('city')['age'].transform('rank')

Merge & Join

# Merge (SQL join)
pd.merge(df1, df2, on='key', how='left')

# Concat (stack)
pd.concat([df1, df2])           # Vertical
pd.concat([df1, df2], axis=1)   # Horizontal

# Append (deprecated, use concat)
pd.concat([df1, df2])

Pivot & Melt

# Pivot table
df.pivot(index='date', columns='product', values='sales')

# Pivot with aggregation
pd.pivot_table(df, values='sales', 
               index='date', columns='product',
               aggfunc='sum')

# Melt (unpivot)
df.melt(id_vars=['date'], value_vars=['sales', 'profit'])

Apply Functions

# Apply function to column
df['name_upper'] = df['name'].apply(str.upper)

# Apply custom function
df['age_group'] = df['age'].apply(
    lambda x: 'young' if x < 30 else 'adult'
)

# Apply to entire DataFrame
df.apply(np.sum, axis=0)  # Column-wise
df.apply(np.sum, axis=1)  # Row-wise

# Vectorized (fastest)
df['age_double'] = df['age'] * 2

String Operations

df['name'].str.lower()
df['name'].str.upper()
df['name'].str.contains('Alice')
df['name'].str.replace('A', 'X')
df['email'].str.split('@').str[0]  # Extract username
df['phone'].str.replace(r'\D', '', regex=True)  # Remove non-digits

DateTime

# Parse dates
df['date'] = pd.to_datetime(df['date'])

# Extract components
df['year'] = df['date'].dt.year
df['month'] = df['date'].dt.month
df['day'] = df['date'].dt.day
df['weekday'] = df['date'].dt.day_name()

# Date range
pd.date_range('2024-01-01', periods=10, freq='D')

# Resample (for time series)
df.set_index('date').resample('M')['value'].sum()
