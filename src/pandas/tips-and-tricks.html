Tips and Tricks

Productivity tips, best practices, and useful tricks for working efficiently with Pandas.

Method Chaining

Method chaining creates cleaner, more readable code by chaining multiple operations together.

Using assign()

# Traditional approach
df = pd.read_csv('file.csv')
df['bonus'] = df['salary'] * 0.1
df['total'] = df['salary'] + df['bonus']
df = df[df['total'] > 50000]
df = df.sort_values('total')

# Method chaining (cleaner)
df = (pd.read_csv('file.csv')
      .assign(
          bonus=lambda x: x['salary'] * 0.1,
          total=lambda x: x['salary'] + x['bonus']
      )
      .query('total > 50000')
      .sort_values('total'))

Using pipe()

def add_features(df):
    return df.assign(
        bonus=df['salary'] * 0.1,
        total=df['salary'] * 1.1
    )

def filter_high(df, threshold=50000):
    return df[df['total'] > threshold]

df = (pd.read_csv('file.csv')
      .pipe(add_features)
      .pipe(filter_high, threshold=60000)
      .sort_values('total'))
Tip: Method chaining is especially useful in Jupyter notebooks where you can see the full pipeline on one screen. It also makes it easier to comment out steps during debugging.

Efficient Data Loading

Load Only Needed Columns

# Load specific columns only
df = pd.read_csv('file.csv', usecols=['name', 'salary', 'dept'])

# Load columns by condition
cols = pd.read_csv('file.csv', nrows=0).columns
wanted_cols = [c for c in cols if 'salary' in c.lower()]
df = pd.read_csv('file.csv', usecols=wanted_cols)

Optimize Data Types on Read

# Specify dtypes when reading
df = pd.read_csv('file.csv', dtype={
    'age': 'int8',
    'salary': 'float32',
    'dept': 'category',
    'active': bool
})

# Parse dates on read
df = pd.read_csv('file.csv', parse_dates=['date', 'start_date'])

# Use date_parser for custom formats
df = pd.read_csv('file.csv', 
    parse_dates=['date'],
    date_parser=lambda x: pd.to_datetime(x, format='%d/%m/%Y'))

Read Large Files in Chunks

# Process large files row by row
for chunk in pd.read_csv('large_file.csv', chunksize=10000):
    # Process chunk
    result = chunk.groupby('dept')['salary'].mean()
    # Store or aggregate results

# Or accumulate in list
chunks = []
for chunk in pd.read_csv('large_file.csv', chunksize=10000):
    filtered = chunk[chunk['salary'] > 50000]
    chunks.append(filtered)
df = pd.concat(chunks, ignore_index=True)

Use Efficient File Formats

# Parquet is much faster and smaller than CSV
df = pd.read_parquet('file.parquet')

# Feather is even faster for reading
df = pd.read_feather('file.feather')

# Pickle preserves dtypes
df = pd.read_pickle('file.pkl')

Data Inspection Tips

Quick Overview

# One-liner overview
df.info()  # Types, memory, nulls

# Statistical summary
df.describe()  # Numeric columns
df.describe(include='all')  # All columns

# First/last rows
df.head(3)   # First 3 rows
df.tail(3)   # Last 3 rows

# Sample
df.sample(5)  # 5 random rows

Memory Usage

# Quick memory check
df.memory_usage(deep=True).sum() / (1024**2)  # MB

# Memory by column
df.memory_usage(deep=True).sort_values(ascending=False)

# Compare before/after optimization
before = df.memory_usage(deep=True).sum()
df['col'] = df['col'].astype('category')
after = df.memory_usage(deep=True).sum()
print(f"Saved: {(before - after) / before * 100:.1f}%")

Check for Problems

# Find columns with most nulls
df.isna().sum().sort_values(ascending=False).head(10)

# Find duplicate rows
df.duplicated().sum()

# Find high cardinality columns
df.nunique().sort_values(ascending=False)

# Check for mixed types in object columns
for col in df.select_dtypes('object'):
    types = df[col].apply(type).unique()
    if len(types) > 1:
        print(f"{col}: {types}")

Column Operations

Rename Multiple Columns

# Rename with function
df.rename(columns=str.lower)
df.rename(columns=str.title)
df.rename(columns=lambda x: x.replace(' ', '_'))

# Rename specific columns
df.rename(columns={'old_name': 'new_name', 'old2': 'new2'})

# Direct assignment
df.columns = ['a', 'b', 'c', 'd']

Create Multiple Columns at Once

# Using assign
df = df.assign(
    bonus=lambda x: x['salary'] * 0.1,
    tax=lambda x: x['salary'] * 0.2,
    net=lambda x: x['salary'] - x['bonus'] - x['tax']
)

# Using eval
df.eval('bonus = salary * 0.1', inplace=True)

Reorder Columns

# Get column order
cols = df.columns.tolist()

# Move column to front
cols = ['target'] + [c for c in cols if c != 'target']
df = df[cols]

# Move column to position
cols = df.columns.tolist()
cols.insert(0, cols.pop(cols.index('target')))
df = df[cols]

Row Operations

Filter Rows Efficiently

# Multiple conditions - create mask first
mask = (df['salary'] > 50000) & (df['dept'] == 'Engineering')
df[mask]

# Using query (often more readable)
df.query('salary > 50000 and dept == "Engineering"')

# Using variables in query
threshold = 50000
df.query('salary > @threshold')

# Using isin for multiple values
depts = ['Engineering', 'HR', 'Finance']
df[df['dept'].isin(depts)]
df.query('dept in @depts')

Select Rows by Position

# First/Last N rows
df.head(10)   # First 10
df.tail(10)   # Last 10

# Every Nth row
df.iloc[::10]  # Every 10th row

# Random sample
df.sample(frac=0.1)  # 10% random sample
df.sample(n=100)      # 100 random rows

Aggregation Tricks

Named Aggregation

# Best practice for groupby
df.groupby('dept').agg(
    headcount=('name', 'count'),
    avg_salary=('salary', 'mean'),
    max_salary=('salary', 'max'),
    total_bonus=('bonus', 'sum')
)

Multiple Aggregations per Column

df.groupby('dept').agg({
    'salary': ['mean', 'max', 'min', 'std'],
    'bonus': ['sum', 'mean']
})

Conditional Aggregation

# Count where condition is true
df.groupby('dept')['active'].agg(['sum', 'count'])

# Or use named aggregation
df.groupby('dept').agg(
    active_count=('active', 'sum'),
    total_count=('active', 'count'),
    active_pct=lambda x: x.sum() / len(x) * 100
)

Top N per Group

# Top 3 earners per department
(df.groupby('dept', as_index=False)
   .apply(lambda x: x.nlargest(3, 'salary'))
   .reset_index(drop=True))

String Operations

Common String Transformations

# Clean whitespace
df['name'] = df['name'].str.strip()
df['name'] = df['name'].str.replace(r'\s+', ' ', regex=True)

# Change case
df['name'] = df['name'].str.lower()
df['name'] = df['name'].str.title()

# Extract parts
df['first_name'] = df['name'].str.split().str[0]
df['last_name'] = df['name'].str.split().str[-1]

# Extract with regex
df[['user', 'domain']] = df['email'].str.extract(r'(\w+)@(\w+)')

# Replace patterns
df['code'] = df['code'].str.replace(r'[^\w]', '', regex=True)

String Containment Check

# Check if contains (case-insensitive)
df[df['text'].str.contains('pattern', case=False)]

# Check if matches pattern
df[df['code'].str.match(r'^[A-Z]{3}\d{4}')]

# Multiple patterns
patterns = ['Engineering', 'HR', 'Finance']
df[df['dept'].str.contains('|'.join(patterns))]

DateTime Tips

Quick Date Operations

df['date'] = pd.to_datetime(df['date'])

# Extract components

df['day'] = df['date'].dt.day

Missing Data Handling

Smart Fill Strategies

# Fill with forward/backward fill
df['value'] = df['value'].ffill()  # Forward fill
df['value'] = df['value'].bfill()  # Backward fill

# Fill with interpolation
df['value'] = df['value'].interpolate(method='linear')

# Fill with group mean
df['value'] = df.groupby('category')['value'].transform(
    lambda x: x.fillna(x.mean())
)

# Fill with previous value + offset
df['value'] = df['value'].ffill() + 100

Identify Patterns in Missing Data

# Which columns have missing values
df.isna().sum()[df.isna().sum() > 0]

# Percentage missing per column
(df.isna().mean() * 100).round(2).sort_values(ascending=False)

# Rows with many missing
df[df.isna().sum(axis=1) > 2]

Merge and Join Tips

Map Values Instead of Merge

# For single column lookup, map is faster than merge
lookup = other_df.set_index('key')['value'].to_dict()
df['value'] = df['key'].map(lookup)

# With default for missing
df['value'] = df['key'].map(lookup).fillna('Unknown')

Merge with Validation

# Always validate your merge
result = pd.merge(df1, df2, on='key', how='left', validate='many_to_one')

# Check merge results
result['_merge'] = result.merge(df2, on='key', how='left', indicator=True)['_merge']
print(result['_merge'].value_counts())

Concatenate Efficiently

# Append single row efficiently
new_row = pd.DataFrame({'col1': [value1], 'col2': [value2]})
df = pd.concat([df, new_row], ignore_index=True)

# Or use .loc
df.loc[len(df)] = [value1, value2]

pivoting and Unpivoting

Quick Pivot

# Basic pivot
df.pivot(index='name', columns='year', values='salary')

# Pivot with aggregation
df.pivot_table(index='name', columns='year', values='salary', aggfunc='mean')

Wide to Long

# Convert columns with similar names
pd.wide_to_long(df, stubnames='Q', i='name', j='quarter')

pd.melt(df, id_vars=['name'], var_name='year', value_name='salary')

Long to Wide

df.pivot(index='name', columns='year', values='salary')

# Using unstack
df.set_index(['name', 'year'])['salary'].unstack()

Conditional Logic

np.where for Simple Conditions

# Simple if-else
df['category'] = np.where(df['salary'] > 100000, 'High', 'Low')

# Multiple conditions
df['category'] = np.where(
    df['salary'] > 100000, 'High',
    np.where(df['salary'] > 50000, 'Medium', 'Low')
)

pd.cut for Binning

# Create bins
df['salary_bin'] = pd.cut(
    df['salary'],
    bins=[0, 50000, 100000, 200000],
    labels=['Low', 'Medium', 'High']
)

# With include_lowest
df['score_bin'] = pd.cut(
    df['score'],
    bins=[0, 50, 70, 90, 100],
    labels=['F', 'C', 'B', 'A'],
    include_lowest=True
)

pd.cut vs pd.qcut

# pd.cut: Equal-width bins
pd.cut(df['salary'], bins=4)  # 4 equal-width bins

# pd.qcut: Equal-frequency bins
pd.qcut(df['salary'], q=4)  # 4 quartiles

Ranking and Sorting

Rank with Tiebreakers

# Different tie methods
df['rank_avg'] = df['salary'].rank(method='average')   # Default
df['rank_min'] = df['salary'].rank(method='min')       # Min rank for ties
df['rank_max'] = df['salary'].rank(method='max')       # Max rank for ties
df['rank_first'] = df['salary'].rank(method='first')    # By order of appearance

Sort with Custom Key

# Sort by length
df.sort_values('name', key=lambda x: x.str.len())

# Sort by month name
month_order = ['January', 'February', 'March', 'April']
df['month'] = pd.Categorical(df['month'], categories=month_order, ordered=True)
df.sort_values('month')

Data Cleaning

Clean Column Names

# Remove special characters
df.columns = df.columns.str.replace(r'[^\w]', '', regex=True)

# Replace spaces with underscores
df.columns = df.columns.str.replace(' ', '_')

# Lowercase
df.columns = df.columns.str.lower()

# Strip whitespace
df.columns = df.columns.str.strip()

Remove Duplicates

# Remove duplicate rows
df.drop_duplicates()

# Keep last occurrence
df.drop_duplicates(keep='last')

# Remove duplicates based on column
df.drop_duplicates(subset=['name', 'email'])

Standardize Values

# Standardize case
df['name'] = df['name'].str.title()

# Trim whitespace
df = df.apply(lambda x: x.str.strip() if x.dtype == 'object' else x)

# Remove non-numeric from column
df['phone'] = df['phone'].str.replace(r'[^\d]', '', regex=True)
