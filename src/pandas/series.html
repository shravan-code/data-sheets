Series

Complete reference of all Series methods organized by category. A Series is a one-dimensional labeled array capable of holding any data type.

Creation Methods

pd.Series() - Main Constructor

import pandas as pd
import numpy as np

# From list
s = pd.Series([1, 2, 3, 4, 5])

# From list with custom index
s = pd.Series([1, 2, 3], index=['a', 'b', 'c'])

# From dictionary (keys become index)
s = pd.Series({'Alice': 95000, 'Bob': 72000, 'Carol': 110000})

# From scalar (broadcasts to index)
s = pd.Series(5, index=range(5))

# From numpy array
s = pd.Series(np.array([1, 2, 3]))

# With name
s = pd.Series([1, 2, 3], name='my_series')

# With dtype
s = pd.Series([1, 2, 3], dtype='float32')

Other Creation Methods

# From array
pd.Series.from_array(np.array([1, 2, 3]))

# From dictionary
pd.Series.from_dict({'a': 1, 'b': 2})

Attributes provide information about the Series structure.

s = pd.Series([1, 2, 3, 4, 5], index=['a', 'b', 'c', 'd', 'e'], name='numbers')

s.values              # Underlying numpy array

s.dtype               # Data type

s.nbytes              # Bytes used
s.empty               # True if Series is empty
s.hasnans             # True if any NaN values
s.is_unique           # True if all values are unique
s.is_monotonic        # True if values are monotonic
s.is_monotonic_increasing  # True if increasing
s.is_monotonic_decreasing  # True if decreasing
s.index.name          # Name of index
s.dtype.name          # Name of dtype

Indexing and Selection

Basic Selection

s = pd.Series([10, 20, 30, 40, 50], index=['a', 'b', 'c', 'd', 'e'])

# By label
s['a']                 # Returns: 10
s['a':'c']             # Returns Series 'a' to 'c' (inclusive)

# By position
s[0]                  # Returns: 10 (first element)
s[0:3]                # Returns first 3 elements

# By list of positions
s[[0, 2, 4]]          # Returns elements at positions 0, 2, 4

# Boolean indexing
s[s > 25]             # Values greater than 25
s[s.isin([10, 30])]  # Values in list

.loc - Label-Based Selection

s.loc['a']             # Single label
s.loc['a':'c']         # Slice (inclusive on both ends)
s.loc[['a', 'c']]      # List of labels
s.loc[s > 25]         # Boolean mask

.iloc - Position-Based Selection

s.iloc[0]             # Single position
s.iloc[0:3]           # Slice (exclusive end, like Python)
s.iloc[[0, 2, 4]]     # List of positions
s.iloc[-1]            # Last element
s.iloc[::2]           # Every other element

Fast Scalar Access

s.at['a']             # Single value by label (fastest)
s.iat[0]              # Single value by position (fastest)

Head, Tail, Sample

s.head()              # First 5 elements
s.head(3)             # First 3 elements
s.tail()              # Last 5 elements
s.tail(2)             # Last 2 elements
s.sample()            # 1 random element
s.sample(n=3)         # 3 random elements
s.sample(frac=0.5)    # 50% random sample

Missing Data Methods

Detect Missing Values

s.isna()              # True where NaN/None
s.isnull()            # Alias for isna()
s.notna()            # True where not NaN
s.notnull()          # Alias for notna()

Fill Missing Values

# Fill with value
s.fillna(0)           # Fill with 0
s.fillna(method='ffill')   # Forward fill
s.fillna(method='bfill')   # Backward fill
s.fillna(s.mean())    # Fill with mean
s.fillna({'a': 1, 'c': 3})  # Fill with dict

Drop Missing Values

s.dropna()            # Remove NaN values

Interpolate

s.interpolate()               # Linear interpolation
s.interpolate(method='linear')
s.interpolate(method='polynomial', order=2)
s.interpolate(method='nearest')
s.interpolate(method='time')       # For datetime index
s.interpolate(limit=2)            # Limit consecutive fills

Where and Mask

s.where(s > 0, other=0)     # Replace where False
s.mask(s > 0, other=0)       # Replace where True

Element-wise Operations

map() - Transform Values

s.map({'a': 1, 'b': 2, 'c': 3})

# Using function
s.map(lambda x: x * 2)

# Using Series
s2 = pd.Series({'a': 1, 'b': 2, 'c': 3})
s.map(s2)

# Using 'first' for unmapped values
s.map({'a': 1, 'b': 2}, na_action='ignore')

apply() - Apply Function

s.apply(lambda x: x * 2)
s.apply(np.sqrt)
s.apply(np.ceil)
s.apply(custom_function)

String Methods (s.str)

Vectorized string operations accessed via .str accessor.

Case Operations

s = pd.Series(['alice', 'BOB', 'Carol'])

s.str.upper()                    # ALL CAPS
s.str.lower()                    # all lower
s.str.title()                    # Title Case
s.str.capitalize()               # First char uppercase
s.str.swapcase()                 # Swap case

Contains and Match

s = pd.Series(['alice@example.com', 'bob@test.org', 'invalid'])

s.str.contains('@')              # Contains substring
s.str.startswith('a')            # Starts with
s.str.endswith('.com')           # Ends with
s.str.match(r'^[a-z]+@')         # Regex match from start
s.str.fullmatch('alice')         # Full regex match

Find and Extract

s = pd.Series(['abc123', 'def456'])

s.str.find('a')                  # Position of substring (-1 if not found)
s.str.rfind('a')                 # Position from right
s.str.index('a')                 # Like find but raises error if not found
s.str.count('a')                 # Count occurrences

# Extract with regex
s.str.extract(r'(\w+)(\d+)')      # Returns DataFrame with groups
s.str.extract(r'(\w+)(\d+)', expand=False)  # Returns Series of tuples

# Extract all matches
s.str.extractall(r'(\w+)')       # For multiple matches

# Find all
s.str.findall(r'\d')             # List of all matches

Replace

s.str.replace('old', 'new')      # Replace substring
s.str.replace(r'\s+', '_', regex=True)  # Regex replace
s.str.replace('a', 'A', n=1)       # Replace first occurrence

Split and Join

s = pd.Series(['alice@example.com', 'bob@test.com'])

s.str.split('@')                  # Split to list
s.str.split('@', expand=True)    # Split to DataFrame columns
s.str.split('@', n=1)             # Max 1 split

# Join (for Series of lists)
pd.Series([['a', 'b'], ['c', 'd']]).str.join('-')

Strip and Pad

s = pd.Series(['  hello  ', '\tworld\n'])

s.str.strip()                     # Remove leading/trailing whitespace
s.str.lstrip()                    # Remove leading
s.str.rstrip()                    # Remove trailing
s.str.strip(' \t\n')              # Remove specific chars

# Pad
s.str.pad(10, ' ', 'left')        # Pad left
s.str.pad(10, ' ', 'right')       # Pad right
s.str.center(10, '-')             # Center
s.str.zfill(5)                    # Zero-fill

Length and Checks

s.str.len()                       # Length of each string
s.str.isalpha()                   # All alphabetic
s.str.isdigit()                    # All digits
s.str.isalnum()                    # Alphanumeric
s.str.isspace()                    # All whitespace
s.str.islower()                    # All lowercase
s.str.isupper()                  # All uppercase
s.str.istitle()                   # Title case

Slice and Substring

s = pd.Series(['abcdef', 'ghijkl'])

s.str[0]                          # First character
s.str[0:3]                        # Characters 0 to 2
s.str[-3:]                        # Last 3 characters
s.str.slice(0, 3)                 # Explicit slice

DateTime Methods (s.dt)

Datetime operations accessed via .dt accessor. Requires datetime dtype.

Component Accessors

s = pd.Series(pd.date_range('2024-01-01', periods=5))

s.dt.year              # Year
s.dt.month             # Month (1-12)
s.dt.day               # Day (1-31)
s.dt.hour              # Hour (0-23)
s.dt.minute            # Minute (0-59)
s.dt.second            # Second (0-59)
s.dt.microsecond       # Microsecond
s.dt.nanosecond        # Nanosecond

Date Properties

s.dt.quarter            # Quarter (1-4)
s.dt.dayofweek         # Day number (0=Mon, 6=Sun)
s.dt.day_name()        # Day name ('Monday', etc.)
s.dt.month_name()      # Month name ('January', etc.)
s.dt.dayofyear         # Day of year (1-366)
s.dt.week              # Week number (0-52)
s.dt.weekday           # Weekday (0-6)
s.dt.days_in_month     # Days in month

Boolean Properties

s.dt.is_month_start    # Is first day of month
s.dt.is_month_end      # Is last day of month
s.dt.is_quarter_start  # Is first day of quarter
s.dt.is_quarter_end    # Is last day of quarter
s.dt.is_year_start     # Is first day of year
s.dt.is_year_end       # Is last day of year
s.dt.is_leap_year      # Is leap year

Datetime Methods

s.dt.to_period('M')    # Convert to Period (month)
s.dt.to_period('Q')    # Convert to Period (quarter)
s.dt.to_period('Y')    # Convert to Period (year)
s.dt.to_timestamp()     # Convert to Timestamp

# Format as string
s.dt.strftime('%Y-%m-%d')
s.dt.strftime('%B %d, %Y')

# Normalize to midnight
s.dt.normalize()

# Floor and ceil
s.dt.floor('D')        # Floor to day
s.dt.ceil('H')         # Ceil to hour
s.dt.round('H')        # Round to hour

Category Methods (s.cat)

Categorical operations accessed via .cat accessor. Requires category dtype.

Category Properties

s = pd.Series(['a', 'b', 'a'], dtype='category')

s.cat.categories        # All categories
s.cat.codes             # Integer codes (0, 1, ...)
s.cat.ordered           # True if ordered

Modify Categories

# Rename categories
s.cat.rename_categories({'a': 'A', 'b': 'B'})

# Reorder categories
s.cat.reorder_categories(['b', 'a'], ordered=True)

# Add categories
s.cat.add_categories(['c'])

# Remove categories
s.cat.remove_categories(['c'])

# Remove unused categories
s.cat.remove_unused_categories()

# Set categories
s.cat.set_categories(['a', 'b', 'c'])

Ordered Operations

# Create ordered categorical
s = pd.Series(['B', 'A', 'C'], dtype='category', 
              categories=['A', 'B', 'C'], ordered=True)

# Comparisons work for ordered categories
s[s >= 'B']
