Pandas Basics

Introduction to Pandas: DataFrames, Series, and basic operations.

Import

import pandas as pd
import numpy as np

Create DataFrame

# From dict
df = pd.DataFrame({
    'name': ['Alice', 'Bob', 'Charlie'],
    'age': [25, 30, 35],
    'city': ['NYC', 'LA', 'Chicago']
})

# From CSV
df = pd.read_csv('data.csv')

# From Excel
df = pd.read_excel('data.xlsx')

# Display
df.head()       # First 5 rows
df.tail()       # Last 5 rows
df.shape         # (rows, columns)
df.info()       # Schema and info
df.describe()   # Statistics

Selection

# Select columns
df['name']              # Series
df[['name', 'age']]     # DataFrame

# Select rows
df.iloc[0]              # By position
df.iloc[0:3]            # Slice
df.loc[0]               # By index
df.loc[0:2]             # Slice (inclusive)

# Filter
df[df['age'] > 25]
df[df['city'].isin(['NYC', 'LA'])]

Basic Operations

# Add column
df['birth_year'] = 2024 - df['age']

# Rename columns
df.rename(columns={'name': 'Name', 'age': 'Age'})

# Drop column
df.drop('city', axis=1)

# Sort
df.sort_values('age', ascending=False)

# Statistics
df['age'].mean()    # Average
df['age'].sum()     # Sum
df['age'].min()     # Minimum
df['age'].max()     # Maximum
df['age'].count()   # Count

Handle Missing Values

# Check for nulls
df.isnull()
df.isnull().sum()

# Drop rows with nulls
df.dropna()

# Fill nulls
df.fillna(0)
df['age'].fillna(df['age'].mean())

# Interpolate
df.interpolate()

Export

# To CSV
df.to_csv('output.csv', index=False)

# To Excel
df.to_excel('output.xlsx', index=False)

# To JSON
df.to_json('output.json')
