collections

from collections import deque, Counter, defaultdict

# deque - fast append/pop from both ends
dq = deque([1, 2, 3])
dq.appendleft(0)    # [0, 1, 2, 3]
dq.popleft()       # [1, 2, 3]

# Counter - frequency counting
cnt = Counter(['a', 'b', 'a', 'c'])
cnt.most_common(2)  # [('a', 2), ('b', 1)]

# defaultdict - auto-initialize keys
dd = defaultdict(list)
dd['fruits'].append('apple')  # {'fruits': ['apple']}

itertools

import itertools

# Infinite iterators
count(10)     # 10, 11, 12, ...
cycle([1,2])  # 1, 2, 1, 2, ...

# Combinatorics
list(itertools.permutations([1,2,3], 2))  # [(1,2), (1,3), ...]
list(itertools.combinations([1,2,3], 2))  # [(1,2), (1,3), (2,3)]

# Chain and accumulate
list(itertools.chain([1,2], [3,4]))              # [1,2,3,4]
list(itertools.accumulate([1,2,3,4]))          # [1, 3, 6, 10]

functools

import functools

# lru_cache - memoization
@functools.lru_cache(maxsize=128)
def fib(n):
    return n if n < 2 else fib(n-1) + fib(n-2)

# partial - bind arguments
from functools import partial
square = partial(pow, exp=2)
square(5)  # 25

# reduce - fold
functools.reduce(lambda a, b: a + b, [1, 2, 3, 4])  # 10

re (Regular Expressions)

import re

# Search and find
re.search(r"\d+", "abc123")     # Match object
re.findall(r"\w+", "hello world")  # ['hello', 'world']

# Replace
re.sub(r"\d+", "X", "abc123")   # "abcX"

# Compile for reuse
pattern = re.compile(r"\d+")
pattern.findall("abc123")
