Python performance

Performance

Memory optimization, threading, multiprocessing, and async programming.

Memory: __slots__

# Without __slots__ - each instance has __dict__
class Point:
    def __init__(self, x, y):
        self.x = x
        self.y = y

# With __slots__ - 60% less memory
class Point:
    __slots__ = ['x', 'y']
    def __init__(self, x, y):
        self.x = x
        self.y = y
Tip: Use __slots__ for data classes to save 40-60% memory.

Memory: Generators

# List - loads everything in memory (~80 MB for 10M)
squares = [x**2 for x in range(10_000_000)]

# Generator - O(1) memory (~112 bytes)
squares = (x**2 for x in range(10_000_000))

# Generator function
def count_up_to(n):
    count = 1
    while count <= n:
        yield count
        count += 1

Threading (I/O-bound)

GIL: Threading helps with I/O tasks, NOT CPU tasks. 
import threading

def download(url):
    # Simulated I/O
    import time
    time.sleep(2)
    return f"Downloaded: {url}"

threads = [
    threading.Thread(target=download, args=(url,))
    for url in urls
]
for t in threads: t.start()
for t in threads: t.join()

Multiprocessing (CPU-bound)

from multiprocessing import Pool

def cpu_task(n):
    return sum(i * i for i in range(n))

if __name__ == "__main__":
    with Pool(4) as pool:
        results = pool.map(cpu_task, [10_000_000] * 8)

Async/Await

import asyncio

async def fetch_data(id):
    await asyncio.sleep(1)  # Non-blocking
    return f"Data {id}"

async def main():
    # Run all concurrently
    results = await asyncio.gather(
        fetch_data(1),
        fetch_data(2),
        fetch_data(3)
    )
    print(results)

asyncio.run(main())  # ~1 second, not 3
