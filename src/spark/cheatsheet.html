PySpark Cheat Sheet

Setup

from pyspark.sql import SparkSession
from pyspark.sql.functions import *
from pyspark.sql.window import Window

spark = SparkSession.builder \
    .appName('App') \
    .master('local[*]') \
    .getOrCreate()

spark.stop()

Create DataFrame

# From list
df = spark.createDataFrame([('a', 1), ('b', 2)], ['col1', 'col2'])

# From CSV
df = spark.read.csv('file.csv', header=True, inferSchema=True)

# From Parquet
df = spark.read.parquet('path/')

# From JSON
df = spark.read.json('path/')

# From JDBC
df = spark.read.format('jdbc').options(url='...', dbtable='...').load()

Select & Filter

# Select columns
df.select('col1', 'col2')
df.select(col('col1'), col('col2') * 2)
df.selectExpr('col1', 'col1 * 2 as doubled')

# Add column
df.withColumn('new', col('col1') + 1)
df.withColumnRenamed('old', 'new')
df.drop('col')

# Filter
df.filter(col('x') > 0)
df.filter((col('x') > 0) & (col('y') < 10))
df.filter(col('x').isin([1, 2, 3]))
df.filter(col('x').isNull())
df.filter(col('x').like('abc%'))

Aggregations

df.groupBy('category').agg(
    count('*').alias('count'),
    sum('amount').alias('total'),
    avg('amount').alias('average'),
    min('amount').alias('min'),
    max('amount').alias('max')
)

df.groupBy('category').pivot('quarter').agg(sum('amount'))

Joins

df1.join(df2, on='key', how='inner')     # matching only
df1.join(df2, on='key', how='left')      # all left
df1.join(df2, on='key', how='right')     # all right
df1.join(df2, on='key', how='outer')     # all
df1.join(df2, on='key', how='semi')      # left with match
df1.join(df2, on='key', how='anti')      # left without match
df1.crossJoin(df2)                         # cartesian

# Multiple conditions
df1.join(df2, (df1.key == df2.key) & (df1.x > df2.x))

# Broadcast
df1.join(broadcast(df2), 'key')

Window Functions

from pyspark.sql.functions import row_number, rank, dense_rank, lag, lead

w = Window.partitionBy('category').orderBy(col('amount').desc())

df.withColumn('rn', row_number().over(w))
df.withColumn('rank', rank().over(w))
df.withColumn('prev', lag('amount', 1).over(w))
df.withColumn('next', lead('amount', 1).over(w))

# Running total
w2 = Window.orderBy('date').rowsBetween(Window.unboundedPreceding, Window.currentRow)
df.withColumn('running', sum('amount').over(w2))

UDFs

from pyspark.sql.functions import udf
from pyspark.sql.types import StringType

@udf(returnType=StringType())
def my_func(x):
    return x.upper() if x else None

df.withColumn('new', my_func('col'))
