Spark DataFrame

DataFrame API

Complete reference for PySpark DataFrame operations.

Selection

from pyspark.sql.functions import col

# Select columns
df.select("name", "age")
df.select(col("name"), col("age"))

# Select with expression
df.select(col("name"), (col("age") * 2).alias("age_doubled"))

# All columns
df.select("*")

# Drop columns
df.drop("unnecessary")

# Select distinct
df.select("city").distinct()

Filtering

# Filter rows
df.filter(col("age") > 25)
df.where("age > 25")
df.filter((col("age") > 25) & (col("city") == "NYC"))
df.filter(col("name").isNull())
df.filter(col("name").isNotNull())

# Multiple conditions
df.filter(col("age").between(20, 30))
df.filter(col("city").isin(["NYC", "LA", "Chicago"]))

Add/Modify Columns

# Add column
df.withColumn("age_doubled", col("age") * 2)
df.withColumn("year", year(col("date")))

# Rename column
df.withColumnRenamed("old_name", "new_name")

# Multiple changes
df.select(
    col("name"),
    col("age"),
    (col("age") * 2).alias("age_doubled")
)

# lit - add constant
df.withColumn("constant", lit(1))

Aggregations

from pyspark.sql.functions import sum, avg, count, min, max

# Basic aggregations
df.agg(sum("salary"), avg("salary"), count("*"))

# Group by
df.groupBy("department").agg(
    sum("salary").alias("total_salary"),
    avg("salary").alias("avg_salary"),
    count("*").alias("employee_count"),
    min("salary").alias("min_salary"),
    max("salary").alias("max_salary")
)

Join

# Join types
df1.join(df2, "key")                    # Inner
df1.join(df2, "key", "left")            # Left
df1.join(df2, "key", "right")           # Right
df1.join(df2, "key", "outer")           # Outer
df1.join(df2, ["key1", "key2"])         # Multiple keys

# Join with condition
df1.join(df2, col("a.id") == col("b.id"), "left")

# Select columns after join
df1.join(df2, "key").select(
    df1["*"], df2["other_column"]
)

Window Functions

from pyspark.sql.window import Window
import pyspark.sql.functions as F

# Define window
window = Window.orderBy("age")
window_partition = Window.partitionBy("department").orderBy("salary")

# Ranking
df.withColumn("rank", F.rank().over(window_partition))
df.withColumn("row_number", F.row_number().over(window_partition))
df.withColumn("dense_rank", F.dense_rank().over(window_partition))

# Running total
df.withColumn("running_total", F.sum("salary").over(window_partition))

# Lag/Lead
df.withColumn("prev_salary", F.lag("salary", 1).over(window_partition))
df.withColumn("next_salary", F.lead("salary", 1).over(window_partition))

Writing

# Write modes
df.write.mode("overwrite").csv("output.csv", header=True)
df.write.mode("append").parquet("output.parquet")
df.write.mode("ignore").saveAsTable("my_table")

# Formats
df.write.csv("output.csv", header=True)
df.write.parquet("output.parquet")
df.write.orc("output.orc")
df.write.mode("overwrite").saveAsTable("table_name")

# Partitioning
df.write.partitionBy("year", "month").parquet("output/")
