Spark Performance Tuning

Caching & Persistence

Cache DataFrames that are reused multiple times to avoid recomputation.

from pyspark import StorageLevel

# Default caching (MEMORY_AND_DISK_DESER)
df.cache()
df.persist()

# Explicit storage levels
df.persist(StorageLevel.MEMORY_ONLY)           # fastest, OOM risk
df.persist(StorageLevel.MEMORY_ONLY_SER)       # serialized
df.persist(StorageLevel.MEMORY_AND_DISK)       # spills to disk if RAM full
df.persist(StorageLevel.MEMORY_AND_DISK_SER)   # serialized + disk
df.persist(StorageLevel.DISK_ONLY)             # no RAM used
df.persist(StorageLevel.OFF_HEAP)             # Tungsten off-heap

# Check if cached
print(df.is_cached)

# Remove from cache
df.unpersist()

Broadcast Joins

For small tables, broadcast them to avoid expensive shuffle operations.

from pyspark.sql.functions import broadcast

# Force broadcast join
result = large_df.join(broadcast(small_df), on='key')

# Broadcast hint in SQL
spark.sql('''
    SELECT /*+ BROADCAST(d) */ *
    FROM orders o
    JOIN departments d ON o.dept_id = d.id
''')
Rule of thumb: Broadcast tables smaller than ~10 MB. Enable spark.sql.autoBroadcastJoinThreshold to let Spark decide automatically.

Partitioning

# Check current partitions
print(df.rdd.getNumPartitions())

# Repartition (full shuffle)
df.repartition(100)                  # 100 partitions
df.repartition(50, 'dept')           # hash partition by dept
df.repartition(8, 'dept', 'year')    # multi-column hash

# Coalesce (reduce without shuffle - faster!)
df.coalesce(10)

# Repartition by range
df.repartitionByRange(20, 'salary')

# Write with partition columns
df.write.mode('overwrite') \
    .partitionBy('year', 'month') \
    .parquet('s3://bucket/output/')

Adaptive Query Execution (AQE)

AQE dynamically re-optimizes queries based on runtime statistics (Spark 3.0+).

# Enable AQE
spark.conf.set('spark.sql.adaptive.enabled', 'true')
spark.conf.set('spark.sql.adaptive.coalescePartitions.enabled', 'true')

# AQE features
spark.conf.set('spark.sql.adaptive.skewJoin.enabled', 'true')    # Handle skewed joins
spark.conf.set('spark.sql.adaptive.autoBroadcastJoinThreshold', '100MB')  # Auto broadcast

Data Skew Handling

# Problem: Some keys have much more data than others

# Solution 1: Broadcast the small side of skewed joins
df1.join(broadcast(df2_skewed), 'key')

# Solution 2: Salting (add random prefix to skewed keys)
from pyspark.sql.functions import concat, lit, rand

salt_count = 10
df_salted = df_skewed.withColumn(
    'salt_key', 
    concat('key', lit('_'), (rand(seed=42) * salt_count).cast('int'))
)

# Join and aggregate results
df_salted.groupBy('key').agg(sum('value')).show()

Serialization

# Use Kryo serializer (faster than default Java)
spark = SparkSession.builder \
    .config('spark.serializer', 'org.apache.spark.serializer.KryoSerializer') \
    .config('spark.kryo.registrationRequired', 'false') \
    .getOrCreate()

# Off-heap memory
spark.conf.set('spark.memory.offHeap.enabled', 'true')
spark.conf.set('spark.memory.offHeap.size', '4g')

Filter Early

# BAD: Load all data, then filter
df = spark.read.parquet('s3://bucket/large-data/')
df_filtered = df.filter(col('date') == '2024-01-01')

# GOOD: Use partition pruning
df_filtered = spark.read.parquet('s3://bucket/large-data/', 
                                  filter=col('date') == '2024-01-01')

Writing Optimized Data

# Coalesce before writing (reduce small files)
df.coalesce(1).write.mode('overwrite').parquet('output/')

# Bucketing (pre-organize data for joins)
df.write.mode('overwrite') \
    .bucketBy(16, 'dept') \
    .sortBy('salary') \
    .saveAsTable('employees_bucketed')

# Z-ordering (optimize filtering on columns)
spark.sql("""
    OPTIMIZE delta.`s3://bucket/table` 
    ZORDER BY (date, region)
""")
