Spark SQL

SQL Interface

Spark provides a full SQL interface. You can register DataFrames as views and query them with SQL.

# Register DataFrame as temp view
df.createOrReplaceTempView('employees')

# Run SQL queries
result = spark.sql('''
    SELECT 
        dept,
        COUNT(*) AS headcount,
        ROUND(AVG(salary), 2) AS avg_salary,
        MAX(salary) AS top_salary
    FROM employees
    WHERE active = true
    GROUP BY dept
    ORDER BY avg_salary DESC
''')
result.show()

# Global temp view (cross-session)
df.createOrReplaceGlobalTempView('global_employees')
spark.sql('SELECT * FROM global_temp.global_employees').show()

Database Operations

# Create database
spark.sql('CREATE DATABASE IF NOT EXISTS hr_db')

# Use database
spark.sql('USE hr_db')

# Show databases
spark.sql('SHOW DATABASES').show()

# Show tables
spark.sql('SHOW TABLES IN hr_db').show()

# Describe table
spark.sql('DESCRIBE TABLE hr_db.employees').show()
spark.sql('DESCRIBE EXTENDED hr_db.employees').show()

Creating Tables

# Create managed table
spark.sql('''
    CREATE TABLE IF NOT EXISTS hr_db.employees (
        name    STRING,
        age     INT,
        dept    STRING,
        salary  DOUBLE
    ) 
    USING PARQUET
    COMMENT 'Employee table'
''')

# Create external table
spark.sql('''
    CREATE EXTERNAL TABLE IF NOT EXISTS hr_db.emp_ext
    USING PARQUET
    LOCATION 's3://bucket/employees/'
''')

# CTAS (Create Table As Select)
spark.sql('''
    CREATE TABLE hr_db.high_earners AS
    SELECT * FROM employees WHERE salary > 90000
''')

Window Functions in SQL

spark.sql('''
    SELECT 
        name,
        dept,
        salary,
        ROW_NUMBER() OVER (PARTITION BY dept ORDER BY salary DESC) as rn,
        RANK() OVER (PARTITION BY dept ORDER BY salary DESC) as rank,
        DENSE_RANK() OVER (PARTITION BY dept ORDER BY salary DESC) as dense_rank,
        LAG(salary, 1) OVER (PARTITION BY dept ORDER BY hire_date) as prev_salary,
        LEAD(salary, 1) OVER (PARTITION BY dept ORDER BY hire_date) as next_salary,
        SUM(salary) OVER (PARTITION BY dept) as dept_total,
        AVG(salary) OVER (PARTITION BY dept) as dept_avg
    FROM employees
''').show()

Complex SQL Operations

# Common Table Expression (CTE)
spark.sql('''
    WITH dept_stats AS (
        SELECT dept, AVG(salary) as avg_sal
        FROM employees
        GROUP BY dept
    )
    SELECT e.name, e.salary, d.avg_sal,
           e.salary - d.avg_sal as diff_from_avg
    FROM employees e
    JOIN dept_stats d ON e.dept = d.dept
''').show()

# Subquery
spark.sql('''
    SELECT name, salary
    FROM employees
    WHERE salary > (SELECT AVG(salary) FROM employees)
''').show()

# UNION / INTERSECT / EXCEPT
spark.sql('SELECT * FROM t1 UNION SELECT * FROM t2')
spark.sql('SELECT * FROM t1 INTERSECT SELECT * FROM t2')
spark.sql('SELECT * FROM t1 EXCEPT SELECT * FROM t2')

Catalog API

# List databases
spark.catalog.listDatabases()

# List tables in database
spark.catalog.listTables('hr_db')

# List columns
spark.catalog.listColumns('hr_db', 'employees')

# Check if cached
spark.catalog.isCached('employees')

# Cache table
spark.catalog.cacheTable('employees')

# Uncache
spark.catalog.uncacheTable('employees')

# Drop temp view
spark.catalog.dropTempView('employees')
spark.catalog.dropGlobalTempView('global_employees')

# Refresh table metadata
spark.catalog.refreshTable('employees')

Query Plans

# Explain query plan
df.explain()                      # Physical plan
df.explain(True)                  # Extended (all plans)
df.explain('cost')                # With cost estimates

# Analyze DataFrame
df.filter(col('salary') > 50000) \
  .groupBy('dept') \
  .agg(avg('salary')) \
  .explain()
Tip: Always check the query plan for large queries to ensure joins are properly broadcast and partitions are reasonable.
