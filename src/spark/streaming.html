Structured Streaming

Structured Streaming treats live data as an unbounded table growing over time. You query it with the same DataFrame/SQL API as batch.

Key Concepts

  • Event Time: When events actually occurred
  • Processing Time: When Spark processes the events
  • Watermark: How late events can arrive and still be included
  • Trigger: When to check for new data
  • Checkpoint: Where to resume after failure

Streaming Architecture

+------------------+     +------------------+     +------------------+
|   INPUT SOURCE   | --> |    SPARK         | --> |   OUTPUT SINK    |
| (Kafka, Files,   |     |    ENGINE        |     | (Delta, Kafka,   |
|  Socket, etc.)   |     | (Micro-batches)  |     |  Console, etc.)  |
+------------------+     +------------------+     +------------------+
         |                        |
         v                        v
+------------------+     +------------------+
|   Checkpoint     |     |  State Store     |
| (Fault Tolerance)|     | (Windowed Aggs)  |
+------------------+     +------------------+

Reading from Kafka

from pyspark.sql import SparkSession
from pyspark.sql import functions as F
from pyspark.sql.types import *

spark = SparkSession.builder.appName('streaming').getOrCreate()

# Kafka source
kafka_df = spark.readStream \
    .format('kafka') \
    .option('kafka.bootstrap.servers', 'broker:9092') \
    .option('subscribe', 'events-topic') \
    .option('startingOffsets', 'latest') \
    .load()

# Kafka schema: key, value, topic, partition, offset, timestamp
raw = kafka_df.select(
    F.from_json(F.col('value').cast('string'), event_schema).alias('data')
).select('data.*')

Window Aggregation

from pyspark.sql.functions import window

# Tumbling window (non-overlapping)
windowed = events \
    .withWatermark('event_time', '10 minutes') \
    .groupBy(
        window(col('event_time'), '5 minutes'),
        'category'
    ).agg(
        count('order_id').alias('order_count'),
        sum('amount').alias('total_revenue')
    )

# Select from window struct
windowed.select(
    col('window.start').alias('window_start'),
    col('window.end').alias('window_end'),
    'category', 'order_count', 'total_revenue'
).show()

Writing to Console

query = windowed.writeStream \
    .outputMode('update') \
    .format('console') \
    .option('truncate', False) \
    .trigger(processingTime='30 seconds') \
    .start()

query.awaitTermination()

Writing to Delta Lake

# Write to Delta table
query = windowed.writeStream \
    .format('delta') \
    .outputMode('update') \
    .option('checkpointLocation', 's3://checkpoints/windowed/') \
    .trigger(processingTime='1 minute') \
    .table('streaming.order_metrics')

# Write to Delta path
query = windowed.writeStream \
    .format('delta') \
    .outputMode('append') \
    .option('checkpointLocation', 's3://checkpoints/delta/') \
    .option('path', 's3://bucket/delta/output/') \
    .start()

ForeachBatch (Custom Sink)

def write_to_database(batch_df, batch_id):
    if batch_df.isEmpty():
        return
    # Write each batch to database
    batch_df.write \
        .mode('append') \
        .jdbc(url, 'table_name', properties)

query = stream_df.writeStream \
    .foreachBatch(write_to_database) \
    .option('checkpointLocation', 's3://checkpoints/foreach/') \
    .trigger(processingTime='1 minute') \
    .start()

Deduplication

# Stateful deduplication within watermark window
deduped = stream_df \
    .withWatermark('event_time', '1 hour') \
    .dropDuplicatesWithinWatermark(['event_id'])

Triggers

# Micro-batch (default)
.writeStream.trigger(processingTime='30 seconds')

# Fixed interval micro-batch
.writeStream.trigger(continuous='10 seconds')  # Spark 3.x

# Process all available data once, then stop
.writeStream.trigger(availableNow=True)

# One-time micro-batch
.writeStream.trigger(once=True)

Streaming Joins

# Stream-static join
stream_df.join(static_df, on='key', how='inner').writeStream

# Stream-stream join with watermark
result = stream1 \
    .withWatermark('left_time', '30 minutes') \
    .join(
        stream2.withWatermark('right_time', '1 hour'),
        on=(stream1.key == stream2.key) & 
            (stream1.left_time >= stream2.right_time - expr('INTERVAL 30 MINUTES')) &
            (stream1.left_time <= stream2.right_time + expr('INTERVAL 30 MINUTES')),
        how='inner'
    )
