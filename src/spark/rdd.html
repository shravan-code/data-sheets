RDD API

Note: RDD is the low-level API. DataFrame API is recommended for most use cases due to better optimization via Catalyst.

Creating RDDs

sc = spark.sparkContext

# From list
rdd = sc.parallelize([1, 2, 3, 4, 5], numSlices=4)

# From text file
rdd = sc.textFile('s3://bucket/logs/*.log')

# Whole text files (filename, content)
rdd = sc.wholeTextFiles('s3://bucket/files/')

# Range
rdd = sc.range(0, 100, step=2)  # 0, 2, 4, ...

# Empty RDD
rdd = sc.emptyRDD()

# From DataFrame
rdd = df.rdd

Basic Transformations

# Map - transform each element
rdd.map(lambda x: x * 2).collect()

# FlatMap - map + flatten
rdd.flatMap(lambda x: [x, x + 1]).collect()

# Filter
rdd.filter(lambda x: x % 2 == 0).collect()

# Distinct
rdd.distinct().collect()

# Sample
rdd.sample(withReplacement=False, fraction=0.5, seed=42).collect()

# Union, Intersection, Subtract
rdd1.union(rdd2).collect()
rdd1.intersection(rdd2).collect()
rdd1.subtract(rdd2).collect()

Pair RDD (Key-Value)

# Create pair RDD
pairs = rdd.map(lambda x: (x % 3, x))  # (key, value)

# Group by key
pairs.groupByKey().collect()

# Reduce by key
pairs.reduceByKey(lambda a, b: a + b).collect()

# Count by key
pairs.countByKey()

# Sort by key
pairs.sortByKey().collect()

# Join
rdd1 = sc.parallelize([('a', 1), ('b', 2)])
rdd2 = sc.parallelize([('a', 10), ('b', 20)])
rdd1.join(rdd2).collect()  # [('a', (1, 10)), ('b', (2, 20))]

Actions

# Collect (return to driver)
rdd.collect()

# Take
rdd.take(5)       # first 5
rdd.first()       # first element
rdd.top(5)        # top 5

# Count
rdd.count()
rdd.countByValue()  # counts each unique value

# Reduce
rdd.reduce(lambda a, b: a + b)

# Fold (like reduce but with initial value)
rdd.fold(0, lambda a, b: a + b)

# Aggregate
rdd.aggregate(
    (0, 0),                           # initial (value, count)
    lambda acc, val: (acc[0] + val, acc[1] + 1),  # seq op
    lambda acc1, acc2: (acc1[0] + acc2[0], acc1[1] + acc2[1])  # comb op
)

Partitioning

# Check partitions
rdd.getNumPartitions()

# Repartition
rdd.repartition(10)

# Coalesce (reduce without shuffle)
rdd.coalesce(4)

# MapPartitions (per partition)
def process_partition(partition):
    return [x * 2 for x in partition]

rdd.mapPartitions(process_partition).collect()

# MapPartitionsWithIndex
def process(index, partition):
    return [(index, x) for x in partition]

rdd.mapPartitionsWithIndex(process).collect()

Broadcast Variables

Share large read-only data efficiently with all executors.

# Broadcast lookup table
lookup = {'Engineering': 'Tech', 'Marketing': 'Biz', 'HR': 'People'}
broadcast_lookup = sc.broadcast(lookup)

# Use in RDD operations
def get_division(dept):
    return broadcast_lookup.value.get(dept, 'Unknown')

df.rdd.map(lambda row: (row.name, get_division(row.dept))).toDF(['name', 'division'])

# Clean up
broadcast_lookup.destroy()
broadcast_lookup.unpersist()

Accumulators

Distributed counters that workers can add to but only driver can read.

# Basic accumulator
error_count = sc.accumulator(0)
null_count = sc.accumulator(0)

def count_errors(row):
    if row.salary is None:
        error_count.add(1)
    if row.name is None:
        null_count.add(1)
    return row

df.rdd.foreach(count_errors)

print(f'Errors: {error_count.value}')
print(f'Nulls: {null_count.value}')

# Custom accumulator (for sets)
from pyspark import AccumulatorParam
class SetAccumulatorParam(AccumulatorParam):
    def zero(self, init): return init
    def addInPlace(self, v1, v2): v1.update(v2); return v1

dept_set = sc.accumulator(set(), SetAccumulatorParam())
df.rdd.foreach(lambda row: dept_set.add({row.dept}))
print(f'Departments: {dept_set.value}')

Pipe Operations

# Pipe through shell commands
rdd.pipe('wc -l').collect()

# More examples
rdd.pipe('grep "ERROR"').collect()
rdd.pipe('awk -F, "{print $1}"').collect()

Glom

# Collect each partition as array
rdd.glom().collect()
# [[1, 2], [3, 4], [5]]  (depends on number of partitions)
