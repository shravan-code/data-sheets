PySpark ETL Pipelines

What is ETL?

ETL stands for Extract, Transform, Load - a data engineering pattern that:

  • Extract: Read data from sources (databases, files, APIs)
  • Transform: Clean, validate, enrich, aggregate data
  • Load: Write processed data to destination

Batch vs Streaming

AspectBatch ETLStreaming
LatencyMinutes to hoursSeconds to milliseconds
TriggerSchedule (cron)Event arrival
VolumeLarge historicalContinuous

Basic ETL Pattern

from pyspark.sql import SparkSession
from pyspark.sql import functions as F
from delta.tables import DeltaTable

spark = SparkSession.builder \
    .appName('etl_pipeline') \
    .config('spark.sql.adaptive.enabled', 'true') \
    .getOrCreate()

RAW_PATH = 's3://bucket/raw/'
OUTPUT_PATH = 's3://bucket/curated/'

# EXTRACT
df = spark.read.parquet(f'{RAW_PATH}/dt=2024-01-15/')

# TRANSFORM
result = df \
    .filter(F.col('status') == 'COMPLETED') \
    .withColumn('revenue', F.col('quantity') * F.col('unit_price')) \
    .withColumn('year', F.year('order_date')) \
    .withColumn('month', F.month('order_date'))

# LOAD
result.write \
    .mode('overwrite') \
    .partitionBy('year', 'month') \
    .parquet(OUTPUT_PATH)

Incremental Load with Delta

def incremental_load(new_data):
    if DeltaTable.isDeltaTable(spark, OUTPUT_PATH):
        delta = DeltaTable.forPath(spark, OUTPUT_PATH)
        
        # Merge (upsert)
        delta.alias('t').merge(
            new_data.alias('s'),
            't.order_id = s.order_id'
        ).whenMatchedUpdateAll() \
         .whenNotMatchedInsertAll() \
         .execute()
    else:
        # First run - create table
        new_data.write.format('delta') \
            .partitionBy('year', 'month') \
            .save(OUTPUT_PATH)

SCD Type 2 Implementation

SCD Type 2 tracks historical changes by creating new versions of records.

from delta.tables import DeltaTable

def apply_scd2(source_df):
    today = F.lit(F.current_date())
    far_future = F.lit('9999-12-31').cast('date')
    
    # Add hash for change detection
    source = source_df.withColumn('record_hash',
        F.md5(F.concat_ws('|', 'name', 'email', 'address')))
    
    if not DeltaTable.isDeltaTable(spark, DIM_PATH):
        # First load
        source.withColumn('valid_from', today) \
              .withColumn('valid_to', far_future) \
              .withColumn('is_current', F.lit(True)) \
              .write.format('delta').save(DIM_PATH)
        return
    
    dim = DeltaTable.forPath(spark, DIM_PATH)
    current = dim.toDF().filter('is_current = true')
    
    # Find changes
    changes = source.alias('s').join(
        current.alias('t'), on='customer_id', how='left'
    ).filter(
        F.col('t.customer_id').isNull() |
        (F.col('s.record_hash') != F.col('t.record_hash'))
    ).select('s.*')
    
    # Expire old, insert new
    dim.alias('target').merge(
        changes.alias('source'),
        'target.customer_id = source.customer_id AND target.is_current = true'
    ).whenMatchedUpdate(set={
        'valid_to': F.current_date(),
        'is_current': F.lit(False)
    }).whenNotMatchedInsertAll().execute()

Data Quality Checks

def validate(df):
    errors = []
    
    # Check required columns
    required_cols = ['order_id', 'customer_id', 'quantity']
    missing = set(required_cols) - set(df.columns)
    if missing:
        errors.append(f'Missing columns: {missing}')
    
    # Check nulls
    null_counts = df.select([
        F.count(F.when(F.col(c).isNull(), c)).alias(c)
        for c in required_cols
    ]).collect()[0].asDict()
    
    for col, count in null_counts.items():
        if count > 0:
            errors.append(f'{col}: {count} null values')
    
    # Check ranges
    if df.filter(F.col('quantity') <= 0).count() > 0:
        errors.append('quantity must be positive')
    
    if errors:
        raise ValueError(f'Data quality errors: {errors}')
    
    return True

validate(df)

Production Best Practices

  • Idempotency: Same input produces same output - safe to rerun
  • Partitioning: Partition output by date/key for efficient reads
  • Schema enforcement: Use explicit schemas, not inference
  • Checkpointing: Track progress for recovery
  • Logging: Log record counts at each stage
  • Error handling: Handle bad records gracefully
