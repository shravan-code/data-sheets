Threading

Thread Basics

threading.Thread runs code concurrently. Best for I/O-bound tasks. GIL limits CPU-bound parallelism.

import threading
import time

def download(url, delay):
    print(f'Starting {url}')
    time.sleep(delay)  # simulate I/O
    print(f'Done {url}')

# Sequential
start = time.perf_counter()
download('site1.com', 1)
download('site2.com', 1)
print(f'Sequential: {time.perf_counter()-start:.1f}s')

# Threaded
start = time.perf_counter()
t1 = threading.Thread(target=download, args=('site1.com', 1))
t2 = threading.Thread(target=download, args=('site2.com', 1))
t1.start()
t2.start()
t1.join()  # wait for t1
t2.join()  # wait for t2
print(f'Threaded: {time.perf_counter()-start:.1f}s')

# Daemon thread - dies with main program
def background_task():
    while True:
        time.sleep(1)
        print('background tick')

daemon = threading.Thread(target=background_task, daemon=True)
daemon.start()
print('Main thread done')

Thread Synchronization: Lock, RLock, Semaphore, Event

Synchronization primitives prevent race conditions when threads share state.

import threading

# Lock - mutual exclusion
class BankAccount:
    def __init__(self, balance):
        self.balance = balance
        self._lock = threading.Lock()
    
    def deposit(self, amount):
        with self._lock:  # acquire and release automatically
            self.balance += amount
    
    def withdraw(self, amount):
        with self._lock:
            if self.balance >= amount:
                self.balance -= amount
                return True
            return False

account = BankAccount(1000)

def do_deposits():
    for _ in range(100):
        account.deposit(10)

def do_withdrawals():
    for _ in range(100):
        account.withdraw(5)

threads = [threading.Thread(target=do_deposits),
           threading.Thread(target=do_withdrawals)]
for t in threads: t.start()
for t in threads: t.join()
print(f'Final balance: {account.balance}')  # deterministic!

# Event - signal between threads
event = threading.Event()

def waiter():
    print('Waiting for event...')
    event.wait()  # blocks until set
    print('Event received!')

t = threading.Thread(target=waiter)
t.start()
import time; time.sleep(0.1)
event.set()  # signal all waiters
t.join()
Output
Final balance: 1500
Waiting for event...
Event received!

ThreadPoolExecutor

High-level interface for managing a pool of worker threads.

from concurrent.futures import ThreadPoolExecutor, as_completed
import time

def fetch_data(url):
    time.sleep(0.5)  # simulate I/O
    return f'Data from {url}'

urls = [f'http://site{i}.com' for i in range(5)]

# map - submit all, collect in order
with ThreadPoolExecutor(max_workers=3) as executor:
    results = list(executor.map(fetch_data, urls))
    print(results)

# submit + as_completed - get results as they finish
with ThreadPoolExecutor(max_workers=3) as executor:
    futures = {executor.submit(fetch_data, url): url for url in urls}
    for future in as_completed(futures):
        url = futures[future]
        try:
            result = future.result()
            print(f'{url} -> {result}')
        except Exception as e:
            print(f'{url} failed: {e}')

Interview Mastery

Q: What is the GIL and how does it affect threading?

A: The Global Interpreter Lock (GIL) allows only one thread to execute Python bytecode at a time in CPython. This means CPU-bound threads don't actually run in parallel. However, the GIL is released during I/O operations, so threading is effective for I/O-bound tasks. For CPU-bound parallelism, use multiprocessing.

Q: What is a race condition and how do you prevent it?

A: A race condition occurs when multiple threads access and modify shared data concurrently, producing unpredictable results. Prevent with: Lock (mutex), RLock (reentrant lock), Semaphore (limit concurrent access), Queue (thread-safe communication), or using thread-local storage.
