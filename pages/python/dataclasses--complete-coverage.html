Section 5: Foundations

Dataclasses — Complete Coverage

@dataclass decorator and all parameters

Dataclasses auto-generate __init__, __repr__, __eq__ and more.

from dataclasses import dataclass, field, fields, asdict, astuple, replace, make_dataclass, is_dataclass
from typing import ClassVar, List

@dataclass(order=True, frozen=False)
class Employee:
    # ClassVar - not included in __init__
    company: ClassVar[str] = 'Acme Corp'
    
    name: str
    department: str
    salary: float
    skills: List[str] = field(default_factory=list)
    _id: int = field(default=0, repr=False, compare=False)
    
    def __post_init__(self):
        if self.salary < 0:
            raise ValueError('Salary cannot be negative')
        self.name = self.name.strip().title()

e1 = Employee('alice', 'Engineering', 90000.0, ['Python', 'SQL'])
e2 = Employee('Bob', 'Marketing', 75000.0)

print(e1)
print(e1.company)
print(e1 > e2)           # order=True compares all fields
print(asdict(e1))
print(astuple(e1))
e3 = replace(e1, salary=95000.0)
print(e3)
print(fields(Employee))  # Field objects
Output
Employee(name='Alice', department='Engineering', salary=90000.0, skills=['Python', 'SQL'])
Acme Corp
True
{'name': 'Alice', 'department': 'Engineering', 'salary': 90000.0, 'skills': ['Python', 'SQL']}
('Alice', 'Engineering', 90000.0, ['Python', 'SQL'], 0)
Employee(name='Alice', department='Engineering', salary=95000.0, skills=['Python', 'SQL'])
(Field(name='name',...), ...)

field() — All Parameters

field() gives fine-grained control over each field.

from dataclasses import dataclass, field

@dataclass
class Config:
    host: str = field(default='localhost', metadata={'help': 'Server host'})
    port: int = field(default=8080)
    tags: list = field(default_factory=list)  # mutable default!
    debug: bool = field(default=False, repr=True, compare=True)
    _internal: str = field(default='', init=False, repr=False)  # not in __init__

c1 = Config()
c2 = Config(host='prod.server.com', port=443, tags=['web'])
print(c1)
print(c2)
print(c1 == Config())  # True

# Inspect field metadata
from dataclasses import fields
for f in fields(Config):
    print(f'{f.name}: default={f.default}, metadata={f.metadata}')
Output
Config(host='localhost', port=8080, tags=[], debug=False)
Config(host='prod.server.com', port=443, tags=['web'], debug=False)
True
host: default=localhost, metadata={'help': 'Server host'}
port: default=8080, metadata={}

Frozen Dataclasses & Inheritance

frozen=True makes instances immutable and hashable.

from dataclasses import dataclass

@dataclass(frozen=True)
class Point:
    x: float
    y: float
    
    def distance(self):
        return (self.x**2 + self.y**2) ** 0.5

p = Point(3.0, 4.0)
print(p.distance())  # 5.0

# Can use as dict key (hashable)
distances = {Point(0,0): 'origin', Point(3,4): 'far'}
print(distances[Point(3,4)])

# try: p.x = 10  -> FrozenInstanceError!

# Inheritance
@dataclass
class Point3D(Point):
    z: float = 0.0

p3 = Point3D(1.0, 2.0, 3.0)
print(p3)
Output
5.0
far
Point3D(x=1.0, y=2.0, z=3.0)

InitVar and __post_init__

InitVar fields are passed to __init__ but not stored as attributes.

from dataclasses import dataclass, field
from dataclasses import InitVar

@dataclass
class DatabaseConnection:
    host: str
    port: int
    password: InitVar[str]  # passed to __post_init__ but not stored
    _connected: bool = field(default=False, init=False, repr=False)
    _conn_string: str = field(default='', init=False, repr=False)
    
    def __post_init__(self, password: str):
        # password available here but not stored as attribute
        self._conn_string = f'{self.host}:{self.port}?pwd={password}'
        print(f'Connecting to {self.host}:{self.port}')

db = DatabaseConnection('localhost', 5432, 'secret123')
print(db)
print(hasattr(db, 'password'))  # False!
print(db._conn_string)
Output
Connecting to localhost:5432
DatabaseConnection(host='localhost', port=5432)
False
localhost:5432?pwd=secret123

All Helper Functions

dataclasses module utility functions.

from dataclasses import (
    dataclass, fields, asdict, astuple,
    replace, make_dataclass, is_dataclass, field
)

@dataclass
class Product:
    name: str
    price: float
    qty: int = 0

p = Product('Apple', 1.5, 100)

# fields() - inspect Field objects
for f in fields(p):
    print(f'{f.name}: type={f.type}, default={f.default}')

# asdict() - recursive dict conversion
print(asdict(p))

# astuple() - recursive tuple conversion
print(astuple(p))

# replace() - copy with changes
p2 = replace(p, price=2.0, qty=50)
print(p2)

# is_dataclass()
print(is_dataclass(Product))   # True
print(is_dataclass(p))         # True
print(is_dataclass(dict))      # False

# make_dataclass() - create dynamically
Point = make_dataclass('Point', ['x', 'y'])
pt = Point(1, 2)
print(pt)

qty: type=int, default=0

Interview Mastery

Q: Why use default_factory instead of a mutable default?

A: Mutable defaults (like `field: list = []`) are shared across all instances — a classic Python bug. `default_factory=list` calls `list()` for each new instance, giving each its own list. This mirrors the mutable default argument trap.

Q: When to use dataclass vs NamedTuple vs plain class?

A: Use dataclass for mutable data with behavior, validation, and flexibility. Use NamedTuple for immutable records that need tuple-like unpacking. Use plain class for complex logic. Dataclasses are the modern default for simple data containers.
