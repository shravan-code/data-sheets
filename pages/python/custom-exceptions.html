Custom Exceptions

Creating Custom Exception Hierarchies

Subclass Exception to create domain-specific errors.

# Base application exception
class AppError(Exception):
    """Base class for all application errors."""
    pass

class ValidationError(AppError):
    def __init__(self, field, message):
        self.field = field
        self.message = message
        super().__init__(f'{field}: {message}')

class DatabaseError(AppError):
    def __init__(self, query, original_error):
        self.query = query
        super().__init__(f'DB error on query "{query}"')
        self.__cause__ = original_error

class NotFoundError(DatabaseError):
    pass

# Usage
try:
    raise ValidationError('email', 'Invalid format')
except ValidationError as e:
    print(f'Field: {e.field}')
    print(f'Error: {e.message}')
except AppError as e:
    print(f'App error: {e}')

# Check hierarchy
err = NotFoundError('SELECT *', Exception('connection refused'))
print(isinstance(err, AppError))      # True
print(isinstance(err, DatabaseError)) # True
Output
Field: email
Error: Invalid format
True
True

Interview Mastery

Q: Why create custom exceptions?

A: Custom exceptions allow callers to catch specific error types from your library without catching unrelated built-in exceptions. They carry domain-specific context (like field name in ValidationError), and enable building clean exception hierarchies that mirror your application's domain.
