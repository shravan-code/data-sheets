Classes & OOP

PowerShell 5+ supports defining full classes with properties, methods, and inheritance.

Defining Classes

Classes allow object-oriented design patterns in PowerShell.

POWERSHELL
class Animal {
    [string]$Name
    [string]$Sound
    [int]$Age

    # Constructor
    Animal([string]$name, [string]$sound) {
        $this.Name  = $name
        $this.Sound = $sound
        $this.Age   = 0
    }

    # Method
    [string] Speak() {
        return "$($this.Name) says: $($this.Sound)!"
    }

    [void] Birthday() {
        $this.Age++
        Write-Host "$($this.Name) is now $($this.Age)"
    }

    # Static method
    static [string] Kingdom() {
        return 'Animalia'
    }
}

# Inheritance
class Dog : Animal {
    [string]$Breed

    Dog([string]$name, [string]$breed) : base($name, 'Woof') {
        $this.Breed = $breed
    }

    [string] Fetch([string]$item) {
        return "$($this.Name) fetches the $item!"
    }
}

# Usage
$cat = [Animal]::new('Whiskers', 'Meow')
$cat.Speak()
$cat.Birthday()
[Animal]::Kingdom()

$dog = [Dog]::new('Rex', 'German Shepherd')
$dog.Speak()
$dog.Fetch('ball')
