Project 1: System Health Monitor

A script that monitors CPU, memory, disk usage, and alerts when thresholds are exceeded.

#!/usr/bin/env bash
set -euo pipefail

# Thresholds
CPU_THRESHOLD=80
MEM_THRESHOLD=80
DISK_THRESHOLD=90

log() { echo "[$(date +'%Y-%m-%d %H:%M:%S')] $1"; }
alert() { echo "[ALERT] $1" >&2; }

check_cpu() {
  local cpu_usage
  cpu_usage=$(top -bn1 | grep "Cpu(s)" | awk '{print $2}' | cut -d'%' -f1 | cut -d'.' -f1)
  log "CPU Usage: ${cpu_usage}%"
  if [ "$cpu_usage" -gt "$CPU_THRESHOLD" ]; then
    alert "CPU usage is ${cpu_usage}% (threshold: ${CPU_THRESHOLD}%)"
  fi
}

check_memory() {
  local mem_usage
  mem_usage=$(free | grep Mem | awk '{print int($3/$2 * 100)}')
  log "Memory Usage: ${mem_usage}%"
  if [ "$mem_usage" -gt "$MEM_THRESHOLD" ]; then
    alert "Memory usage is ${mem_usage}% (threshold: ${MEM_THRESHOLD}%)"
  fi
}

check_disk() {
  while IFS= read -r line; do
    local usage
    local mount
    usage=$(echo "$line" | awk '{print $5}' | tr -d '%')
    mount=$(echo "$line" | awk '{print $6}')
    log "Disk $mount: ${usage}%"
    if [ "$usage" -gt "$DISK_THRESHOLD" ]; then
      alert "Disk $mount is ${usage}% full (threshold: ${DISK_THRESHOLD}%)"
    fi
  done < <(df -h | grep -E '^/dev')
}

log "=== System Health Check ==="
check_cpu
check_memory
check_disk
log "=== Check Complete ==="

Project 2: Automated Backup Script

Backs up files/directories to a timestamped archive with retention policy and logging.

#!/usr/bin/env bash
set -euo pipefail

SOURCE_DIR="${1:-/home/$USER/Documents}"
BACKUP_DIR="${2:-/backup}"
RETENTION_DAYS=7
TIMESTAMP=$(date +%Y%m%d_%H%M%S)
BACKUP_NAME="backup_${TIMESTAMP}.tar.gz"
LOG_FILE="/var/log/backup.log"

log()  { echo "[INFO]  $(date +'%Y-%m-%d %H:%M:%S') $1" | tee -a "$LOG_FILE"; }
error(){ echo "[ERROR] $(date +'%Y-%m-%d %H:%M:%S') $1" | tee -a "$LOG_FILE" >&2; }

create_backup() {
  log "Starting backup of $SOURCE_DIR"
  mkdir -p "$BACKUP_DIR"
  if tar -czf "${BACKUP_DIR}/${BACKUP_NAME}" -C "$(dirname $SOURCE_DIR)" "$(basename $SOURCE_DIR)" 2>>/tmp/tar_err.log; then
    local size
    size=$(du -sh "${BACKUP_DIR}/${BACKUP_NAME}" | cut -f1)
    log "Backup created: ${BACKUP_NAME} (${size})"
  else
    error "Backup failed!"
    exit 1
  fi
}

rotate_backups() {
  log "Rotating backups older than $RETENTION_DAYS days"
  find "$BACKUP_DIR" -name "backup_*.tar.gz" -mtime +$RETENTION_DAYS -delete
  local count
  count=$(find "$BACKUP_DIR" -name "backup_*.tar.gz" | wc -l)
  log "Remaining backups: $count"
}

create_backup
rotate_backups
log "Backup process completed successfully"

Project 3: User Management Script

Adds, removes, lists, and modifies system users with validation and logging.

#!/usr/bin/env bash
set -euo pipefail

[ "$EUID" -ne 0 ] && { echo "Run as root"; exit 1; }

LOG_FILE="/var/log/user_management.log"

log() { echo "[$(date +'%Y-%m-%d %H:%M:%S')] $1" | tee -a "$LOG_FILE"; }

usage() {
  cat << EOF
Usage: $0 <command> [options]
Commands:
  add    <username> [--group <group>] [--shell <shell>]
  remove <username>
  list
  info   <username>
EOF
  exit 1
}

add_user() {
  local username="$1"; shift
  local group="users"
  local shell="/bin/bash"
  while [[ $# -gt 0 ]]; do
    case "$1" in
      --group) group="$2"; shift 2 ;;
      --shell) shell="$2"; shift 2 ;;
      *) echo "Unknown option: $1"; exit 1 ;;
    esac
  done
  id "$username" &>/dev/null && { echo "User $username already exists"; exit 1; }
  useradd -m -s "$shell" -G "$group" "$username"
  log "User $username created (group=$group, shell=$shell)"
  echo "User $username created. Set password with: passwd $username"
}

remove_user() {
  local username="$1"
  id "$username" &>/dev/null || { echo "User $username not found"; exit 1; }
  userdel -r "$username"
  log "User $username removed"
  echo "User $username removed"
}

list_users() {
  echo "=== System Users ==="
  awk -F: '$3 >= 1000 && $1 != "nobody" {printf "%-20s UID:%-8s Shell: %s\n", $1, $3, $7}' /etc/passwd
}

info_user() {
  local username="$1"
  id "$username" &>/dev/null || { echo "User $username not found"; exit 1; }
  id "$username"
  echo "Home: $(eval echo ~$username)"
  echo "Shell: $(getent passwd $username | cut -d: -f7)"
  echo "Groups: $(groups $username)"
}

[ $# -lt 1 ] && usage
command="$1"; shift
case "$command" in
  add)    add_user "$@" ;;
  remove) remove_user "$@" ;;
  list)   list_users ;;
  info)   info_user "$@" ;;
  *)      usage ;;
esac

Project 4: Log Analyzer

Parses web server or system logs to generate summary reports of errors, IPs, and request counts.

#!/usr/bin/env bash
set -euo pipefail

LOG_FILE="${1:-/var/log/nginx/access.log}"
[ -f "$LOG_FILE" ] || { echo "Log file not found: $LOG_FILE"; exit 1; }

echo "===================================="
echo "  Log Analysis Report"
echo "  File: $LOG_FILE"
echo "  Date: $(date)"
echo "===================================="

echo -e "\n--- Total Requests ---"
total=$(wc -l < "$LOG_FILE")
echo "Total lines: $total"

echo -e "\n--- Top 10 IP Addresses ---"
awk '{print $1}' "$LOG_FILE" | sort | uniq -c | sort -rn | head -10 | \
  awk '{printf "  %-8s %s\n", $1, $2}'

echo -e "\n--- HTTP Status Code Distribution ---"
awk '{print $9}' "$LOG_FILE" | sort | uniq -c | sort -rn | \
  awk '{printf "  Status %-6s Count: %s\n", $2, $1}'

echo -e "\n--- Top 10 Requested URLs ---"
awk '{print $7}' "$LOG_FILE" | sort | uniq -c | sort -rn | head -10 | \
  awk '{printf "  %-8s %s\n", $1, $2}'

echo -e "\n--- Errors (4xx and 5xx) ---"
grep -E '" [45][0-9]{2} ' "$LOG_FILE" | wc -l | \
  awk '{print "  Error requests: " $1}'

echo -e "\n--- Requests per Hour ---"
awk '{print $4}' "$LOG_FILE" | cut -d: -f2 | sort | uniq -c | \
  awk '{printf "  Hour %02d:00 - %s requests\n", $2, $1}'

echo -e "\n==================================="

Project 5: Interactive Menu-Driven App

A full interactive menu-driven script for common sysadmin tasks using select and functions.

#!/usr/bin/env bash
set -euo pipefail

# Colors
RED='\033[0;31m'
GREEN='\033[0;32m'
YELLOW='\033[1;33m'
BLUE='\033[0;34m'
NC='\033[0m'  # No Color

print_header() {
  clear
  echo -e "${BLUE}================================${NC}"
  echo -e "${BLUE}  SysAdmin Toolkit v1.0${NC}"
  echo -e "${BLUE}================================${NC}\n"
}

show_system_info() {
  echo -e "${GREEN}=== System Information ===${NC}"
  echo "OS:       $(uname -o)"
  echo "Kernel:   $(uname -r)"
  echo "Hostname: $(hostname)"
  echo "Uptime:   $(uptime -p)"
  echo "Users:    $(who | wc -l) logged in"
  echo "Load:     $(uptime | awk -F'load average:' '{print $2}')"
}

show_disk_usage() {
  echo -e "${GREEN}=== Disk Usage ===${NC}"
  df -h | grep -E '^/dev' | awk '{
    printf "%-20s %6s used of %-6s (%s full)\n", $1, $3, $2, $5
  }'
}

show_memory() {
  echo -e "${GREEN}=== Memory Usage ===${NC}"
  free -h | awk '/^Mem/ {printf "RAM:  %s used / %s total (%.1f%%)\n", $3, $2, $3/$2*100}'
  free -h | awk '/^Swap/ {printf "Swap: %s used / %s total\n", $3, $2}'
}

show_top_processes() {
  echo -e "${GREEN}=== Top 5 Processes by CPU ===${NC}"
  ps aux --sort=-%cpu | head -6 | tail -5 | \
    awk '{printf "%-20s CPU:%-6s MEM:%-6s\n", $11, $3, $4}'
}

network_info() {
  echo -e "${GREEN}=== Network Information ===${NC}"
  ip addr show | grep 'inet ' | awk '{print "IP:", $2}'
  echo "Connections: $(ss -tn | grep ESTAB | wc -l) established"
}

main_menu() {
  local options=("System Info" "Disk Usage" "Memory" "Top Processes" "Network" "Quit")
  while true; do
    print_header
    PS3=$'\nSelect option: '
    select opt in "${options[@]}"; do
      case $opt in
        "System Info")    show_system_info ;;
        "Disk Usage")     show_disk_usage ;;
        "Memory")         show_memory ;;
        "Top Processes")  show_top_processes ;;
        "Network")        network_info ;;
        "Quit")           echo -e "${YELLOW}Goodbye!${NC}"; exit 0 ;;
        *) echo -e "${RED}Invalid option${NC}" ;;
      esac
      echo -e "\nPress Enter to continue..."
      read -r
      break
    done
  done
}
