Databases

sqlite3 and SQLAlchemy

Database access using built-in sqlite3 and SQLAlchemy ORM.

import sqlite3
from contextlib import contextmanager

# sqlite3 - built-in
@contextmanager
def get_db(path=':memory:'):
    conn = sqlite3.connect(path)
    conn.row_factory = sqlite3.Row  # access by column name
    try:
        yield conn
        conn.commit()
    except Exception:
        conn.rollback()
        raise
    finally:
        conn.close()

with get_db() as db:
    # Create table
    db.execute('''CREATE TABLE users (
        id INTEGER PRIMARY KEY AUTOINCREMENT,
        name TEXT NOT NULL,
        email TEXT UNIQUE,
        age INTEGER
    )''')
    
    # Insert with parameterized queries (prevent SQL injection)
    db.execute('INSERT INTO users (name, email, age) VALUES (?,?,?)',
               ('Alice', 'alice@example.com', 30))
    db.executemany('INSERT INTO users (name, email, age) VALUES (?,?,?)',
                   [('Bob', 'bob@example.com', 25),
                    ('Carol', 'carol@example.com', 35)])
    
    # Query
    cursor = db.execute('SELECT * FROM users WHERE age > ?', (24,))
    for row in cursor:
        print(dict(row))
    
    # Aggregate
    result = db.execute('SELECT AVG(age) as avg_age FROM users').fetchone()
    print(f'Avg age: {result["avg_age"]:.1f}')

Interview Mastery

Q:

What is SQL injection and how do you prevent it?

A:

SQL injection is when user input is embedded directly in SQL strings, allowing attackers to modify queries. Prevent by ALWAYS using parameterized queries: `cursor.execute('SELECT * FROM users WHERE id = ?', (user_id,))` — NEVER f-strings or string concatenation for SQL.
