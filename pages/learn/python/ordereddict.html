OrderedDict

OrderedDict Methods

Dict with ordering-aware operations. Less needed since Python 3.7+ dicts preserve order.

from collections import OrderedDict

od = OrderedDict([('a', 1), ('b', 2), ('c', 3)])
print(od)

# move_to_end
od.move_to_end('a')           # move to end
print(od)
od.move_to_end('c', last=False)  # move to front
print(od)

# popitem(last=True) removes from end (LIFO)
key, val = od.popitem(last=True)
print(f'Removed: {key}={val}', od)

# OrderedDict equality checks ORDER
od1 = OrderedDict([('a',1),('b',2)])
od2 = OrderedDict([('b',2),('a',1)])
od3 = OrderedDict([('a',1),('b',2)])
print(od1 == od2)  # False! (different order)
print(od1 == od3)  # True
print(od1 == {'a':1,'b':2})  # True (regular dict ignores order)
