reduce — Fold Iterables

Apply function cumulatively to reduce iterable to single value.

from functools import reduce
import operator

# Sum (prefer sum() built-in, this is for demonstration)
print(reduce(lambda acc, x: acc + x, [1,2,3,4,5]))

# Product
print(reduce(operator.mul, [1,2,3,4,5]))

# With initializer (handles empty list)
print(reduce(operator.add, [], 0))   # 0

# Flatten nested list
nested = [[1,2],[3,4],[5,6]]
flat = reduce(lambda a, b: a + b, nested)
print(flat)

# Find max without max()
print(reduce(lambda a,b: a if a > b else b, [3,1,4,1,5,9,2,6]))
