Infinite Iterators

count

An iterator that produces values by adding a step to a start value indefinitely.

from itertools import count, islice

# count(start, step)
for n in islice(count(10, 3), 5):
    print(n, end=' ')
print()

# count with floats
for f in islice(count(0.0, 0.5), 6):
    print(f, end=' ')
Output
10 13 16 19 22 
0.0 0.5 1.0 1.5 2.0 2.5

cycle

An iterator that repeats the elements of an iterable indefinitely.

from itertools import cycle, islice

colors = cycle(['red', 'green', 'blue'])
for c in islice(colors, 7):
    print(c, end=' ')
Output
red green blue red green blue red

repeat

An iterator that produces the same value either indefinitely or a fixed number of times.

from itertools import repeat

# repeat(val, times)
for val in repeat('hello', 3):
    print(val, end=' ')
print()

# repeat with map
result = list(map(pow, range(5), repeat(2)))
print(result)
Output
hello hello hello 
[0, 1, 4, 9, 16]
