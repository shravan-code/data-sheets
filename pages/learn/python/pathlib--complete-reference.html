pathlib — Complete Reference

Path Object — All Properties & Methods

pathlib provides an OOP interface to file system paths.

from pathlib import Path

# Construction
p = Path('/home/user/documents/report.pdf')

# Properties
print(p.name)       # report.pdf
print(p.stem)       # report
print(p.suffix)     # .pdf
print(p.suffixes)   # ['.pdf']
print(p.parent)     # /home/user/documents
print(p.parents[0]) # /home/user/documents
print(p.parents[1]) # /home/user
print(p.root)       # /
print(p.anchor)     # /
print(p.parts)      # ('/', 'home', 'user', 'documents', 'report.pdf')
print(p.drive)      # '' (empty on Unix)

# Build paths with / operator
base = Path('/home/user')
doc = base / 'documents' / 'report.pdf'
print(doc)

# Modify path
print(p.with_name('summary.pdf'))     # change filename
print(p.with_stem('summary'))         # change stem (3.9+)
print(p.with_suffix('.txt'))          # change extension

Path I/O and Filesystem Operations

Read, write, and navigate the filesystem with Path objects.

from pathlib import Path

# Create and write files
p = Path('/tmp/test_pathlib.txt')
p.write_text('Hello, pathlib!')

# Read
print(p.read_text())

# Exists and type checks
print(p.exists())    # True
print(p.is_file())   # True
print(p.is_dir())    # False

# Stat info
stat = p.stat()
print(f'Size: {stat.st_size} bytes')

# Directory operations
dir_path = Path('/tmp/test_dir')
dir_path.mkdir(exist_ok=True)
(dir_path / 'file1.txt').write_text('File 1')
(dir_path / 'file2.txt').write_text('File 2')
(dir_path / 'subdir').mkdir(exist_ok=True)

# List directory
for item in dir_path.iterdir():
    print(f'{item.name}: {"dir" if item.is_dir() else "file"}')

# Glob patterns
for txt in dir_path.glob('*.txt'):
    print(txt.name)

# Recursive glob
for f in dir_path.rglob('*'):
    print(f.relative_to(dir_path))

# Cleanup
p.unlink()
import shutil
shutil.rmtree(dir_path)

Interview Mastery

Q:

Why use pathlib over os.path?

A:

pathlib provides an OOP interface — paths are objects with methods, not strings. The / operator for joining is cleaner than os.path.join(). Methods like .read_text(), .write_text() simplify file I/O. Path objects work across OS. For new code, pathlib is always preferred.
