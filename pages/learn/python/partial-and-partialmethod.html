partial and partialmethod

partial — Freeze Function Arguments

Create a new callable with some arguments pre-filled.

from functools import partial

def power(base, exponent):
    return base ** exponent

# Freeze exponent
square = partial(power, exponent=2)
cube   = partial(power, exponent=3)

print(square(4))   # 16
print(cube(3))     # 27

# Inspect partial
print(square.func)     # <function power>
print(square.keywords) # {'exponent': 2}

# Practical: adapt functions for map/filter
from functools import partial

def multiply(x, y): return x * y
double = partial(multiply, y=2)
print(list(map(double, [1, 2, 3, 4, 5])))

# With positional args
print_error = partial(print, 'ERROR:', sep='\n', end='\n---\n')
print_error('Connection failed')
