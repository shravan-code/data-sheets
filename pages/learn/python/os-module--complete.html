os Module — Complete

File & Directory Operations

Core file system operations via the os module.

import os

# Current directory
print(os.getcwd())

# List directory
files = os.listdir('.')
print(files[:3])

# scandir - richer than listdir
for entry in os.scandir('.'):
    print(f'{entry.name}: isfile={entry.is_file()}')
    break  # just first entry

# os.walk - recursive tree traversal
for root, dirs, files in os.walk('/tmp', topdown=True):
    print(f'Dir: {root}')
    print(f'  Subdirs: {dirs}')
    print(f'  Files: {files}')
    break  # just root level

# Create / remove directories
os.makedirs('/tmp/test/nested', exist_ok=True)
print(os.path.exists('/tmp/test/nested'))  # True
os.removedirs('/tmp/test/nested')          # removes empty dirs

# File operations
with open('/tmp/test.txt', 'w') as f:
    f.write('hello')
os.rename('/tmp/test.txt', '/tmp/test2.txt')
print(os.path.exists('/tmp/test2.txt'))  # True
os.remove('/tmp/test2.txt')

os.path — All Methods

Path manipulation without creating Path objects.

import os

path = '/home/user/documents/report.pdf'

# Decompose path
print(os.path.dirname(path))     # /home/user/documents
print(os.path.basename(path))    # report.pdf
print(os.path.split(path))       # ('/home/user/documents', 'report.pdf')
print(os.path.splitext(path))    # ('/home/user/documents/report', '.pdf')

# Join paths (OS-safe)
print(os.path.join('/home', 'user', 'file.txt'))

# Check existence
print(os.path.exists('/tmp'))
print(os.path.isfile('/tmp'))
print(os.path.isdir('/tmp'))

# Normalize and resolve
print(os.path.abspath('.'))
print(os.path.normpath('/home/user/../user/./docs'))
print(os.path.expanduser('~'))
print(os.path.expandvars('$HOME/docs'))

# Relative path
print(os.path.relpath('/home/user/docs', '/home/user'))

Environment Variables & Process

Access and manipulate environment variables and process info.

import os

# Environment variables
print(os.environ.get('HOME', '/default'))
print(os.environ.get('NONEXISTENT', 'fallback'))

# Set/unset
os.environ['MY_APP_ENV'] = 'production'
print(os.getenv('MY_APP_ENV'))   # production

# Process info
print(os.getpid())         # current process ID
print(os.cpu_count())      # number of CPUs
print(os.name)             # 'posix' or 'nt'

# System constants
print(repr(os.sep))        # '/' on Unix, '\\' on Windows
print(repr(os.linesep))    # '\n' on Unix, '\r\n' on Windows
print(repr(os.pathsep))    # ':' on Unix, ';' on Windows
print(os.curdir, os.pardir)  # . and ..
print(os.devnull)          # /dev/null
