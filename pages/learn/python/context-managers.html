Context Managers

Context Manager Protocol and contextlib

__enter__ and __exit__ implement the with statement protocol.

import contextlib
import time

# Class-based context manager
class Timer:
    def __enter__(self):
        self.start = time.perf_counter()
        return self  # value bound to 'as' variable
    
    def __exit__(self, exc_type, exc_val, exc_tb):
        self.elapsed = time.perf_counter() - self.start
        print(f'Elapsed: {self.elapsed:.4f}s')
        return False  # don't suppress exceptions

with Timer() as t:
    sum(range(1000000))
print(f'Timer says: {t.elapsed:.4f}s')

# Generator-based with contextlib
@contextlib.contextmanager
def managed_resource(name):
    print(f'Acquiring {name}')
    resource = {'name': name, 'active': True}
    try:
        yield resource  # code in 'with' block runs here
    finally:
        resource['active'] = False
        print(f'Releasing {name}')

with managed_resource('DB Connection') as r:
    print(f'Using {r["name"]}')
print(f'Active: {r["active"]}')

# contextlib.suppress - swallow specific exceptions
with contextlib.suppress(FileNotFoundError):
    open('nonexistent.txt')
print('Continued after suppressed error')

# contextlib.nullcontext - conditional context manager
def process(use_timer=True):
    ctx = Timer() if use_timer else contextlib.nullcontext()
    with ctx:
        return sum(range(100000))

process(use_timer=True)
Output
Elapsed: 0.0234s
Timer says: 0.0234s
Acquiring DB Connection
Using DB Connection
Releasing DB Connection
Active: False
Continued after suppressed error
Elapsed: 0.0012s

Interview Mastery

Q:

What should __exit__ return?

A:

Return True to suppress exceptions raised in the with block. Return False (or None) to propagate them. exc_type, exc_val, exc_tb are None if no exception occurred. The finally clause in @contextmanager always runs, making cleanup guaranteed.
