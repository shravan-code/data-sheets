Multiprocessing

Process and Pool

multiprocessing bypasses the GIL for true CPU parallelism by using separate processes.

from multiprocessing import Process, Pool, Queue, Value, Array
import os

def cpu_work(n):
    # CPU-bound: sum of squares
    return sum(i*i for i in range(n))

# Pool.map - parallel map
if __name__ == '__main__':  # required guard on Windows/macOS
    with Pool(processes=4) as pool:
        results = pool.map(cpu_work, [10**6, 10**6, 10**6, 10**6])
        print(results[:2])
    
    # Pool.starmap - multiple args
    def add(a, b): return a + b
    with Pool(4) as pool:
        results = pool.starmap(add, [(1,2),(3,4),(5,6)])
        print(results)
    
    # Pool.apply_async - non-blocking
    with Pool(2) as pool:
        async_result = pool.apply_async(cpu_work, (10**5,))
        print('Submitted, doing other work...')
        print(async_result.get(timeout=10))

ProcessPoolExecutor and Inter-process Communication

High-level process pool and IPC mechanisms.

from concurrent.futures import ProcessPoolExecutor
from multiprocessing import Queue, Pipe

def heavy_compute(x):
    return x ** 2

# ProcessPoolExecutor (similar to ThreadPoolExecutor API)
if __name__ == '__main__':
    with ProcessPoolExecutor(max_workers=4) as executor:
        results = list(executor.map(heavy_compute, range(10)))
        print(results)

# Queue - safe IPC
def producer(queue):
    for i in range(5):
        queue.put(i)
    queue.put(None)  # sentinel

def consumer(queue):
    while True:
        item = queue.get()
        if item is None:
            break
        print(f'Consumed: {item}')

if __name__ == '__main__':
    q = Queue()
    p = Process(target=producer, args=(q,))
    c = Process(target=consumer, args=(q,))
    p.start(); c.start()
    p.join(); c.join()

Interview Mastery

Q:

When use threading vs multiprocessing?

A:

Threading: I/O-bound tasks (network requests, file operations, DB queries) where threads wait most of the time. Multiprocessing: CPU-bound tasks (number crunching, image processing, ML) where you need true parallelism. Multiprocessing has higher overhead (process creation, IPC) but bypasses the GIL.
