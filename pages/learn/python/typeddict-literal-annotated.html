TypedDict, Literal, Annotated

TypedDict — Typed Dictionaries

Define the expected structure of a dictionary with type safety.

from typing import TypedDict, Required, NotRequired

class UserProfile(TypedDict):
    name: str
    age: int
    email: str

# With optional fields (Python 3.11+)
class Config(TypedDict, total=False):  # all optional
    debug: bool
    timeout: int
    host: str

# Mixed required/optional (Python 3.11+)
class ApiRequest(TypedDict):
    endpoint: str              # required
    method: str                # required
    timeout: NotRequired[int]  # optional

user: UserProfile = {'name': 'Alice', 'age': 30, 'email': 'a@b.com'}
print(user['name'])

request: ApiRequest = {'endpoint': '/api/data', 'method': 'GET'}
print(request)
Output
Alice
{'endpoint': '/api/data', 'method': 'GET'}

Literal, Final, ClassVar, Annotated

Advanced type annotations for precise type constraints.

from typing import Literal, Final, ClassVar, Annotated
from dataclasses import dataclass

# Literal - only specific values allowed
Direction = Literal['north', 'south', 'east', 'west']

def move(direction: Direction, steps: int) -> str:
    return f'Moving {direction} {steps} steps'

print(move('north', 5))

# Final - cannot be reassigned
MAX_RETRIES: Final = 3
API_URL: Final[str] = 'https://api.example.com'

# ClassVar - class-level variable in dataclass/class
@dataclass
class Counter:
    total: ClassVar[int] = 0  # not in __init__
    value: int = 0

c = Counter(10)
print(c.value, Counter.total)

# Annotated - attach metadata
def process(x: Annotated[int, 'must be positive']) -> int:
    return x * 2

from typing import get_type_hints
hints = get_type_hints(process, include_extras=True)
print(hints)
Output
Moving north 5 steps
10 0
{'x': typing.Annotated[int, 'must be positive'], 'return': <class 'int'>}
