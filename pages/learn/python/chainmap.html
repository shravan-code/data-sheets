ChainMap

ChainMap — Layered Lookups

Search through multiple dicts in order. Updates go to first map only.

from collections import ChainMap

# CLI args override env vars override defaults
defaults   = {'color': 'blue', 'user': 'guest', 'timeout': 30}
env_vars   = {'user': 'admin', 'timeout': 60}
cli_args   = {'color': 'red'}

cm = ChainMap(cli_args, env_vars, defaults)

print(cm['color'])    # red   (from cli_args)
print(cm['user'])     # admin (from env_vars)
print(cm['timeout'])  # 60    (from env_vars)

# Update only affects first map
cm['new_key'] = 'value'
print(cli_args)  # {'color':'red','new_key':'value'}

# new_child - create child scope
child = cm.new_child({'color': 'green'})
print(child['color'])  # green (from child)
print(cm['color'])     # red   (original unchanged)

# parents - all maps except first
print(list(cm.parents.maps))
print(list(cm.maps))
