Metaclasses

type and Custom Metaclasses

Classes are objects too. type is the metaclass of all classes. You can customize class creation.

# type is the metaclass of all classes
print(type(int))     # <class 'type'>
print(type(list))    # <class 'type'>
print(type(type))    # <class 'type'> (type is its own metaclass)

# Create class dynamically with type
Dog = type('Dog', (object,), {
    'species': 'Canis lupus',
    'bark': lambda self: 'Woof!'
})
d = Dog()
print(d.bark())
print(Dog.species)

# Custom metaclass
class SingletonMeta(type):
    _instances = {}
    
    def __call__(cls, *args, **kwargs):
        if cls not in cls._instances:
            cls._instances[cls] = super().__call__(*args, **kwargs)
        return cls._instances[cls]

class DatabasePool(metaclass=SingletonMeta):
    def __init__(self):
        self.connections = []

db1 = DatabasePool()
db2 = DatabasePool()
print(db1 is db2)   # True - same instance

# __init_subclass__ - lighter alternative to metaclass
class PluginBase:
    _plugins = {}
    
    def __init_subclass__(cls, plugin_name=None, **kwargs):
        super().__init_subclass__(**kwargs)
        if plugin_name:
            PluginBase._plugins[plugin_name] = cls

class CSVPlugin(PluginBase, plugin_name='csv'):
    pass

class JSONPlugin(PluginBase, plugin_name='json'):
    pass

print(PluginBase._plugins)
Output
<class 'type'>
<class 'type'>
<class 'type'>
Woof!
Canis lupus
True
{'csv': <class 'CSVPlugin'>, 'json': <class 'JSONPlugin'>}

Interview Mastery

Q:

When would you use a metaclass?

A:

Metaclasses are for 'class creation hooks': enforcing interface contracts, registering subclasses automatically (plugin systems), modifying class attributes at definition time, ORMs (like Django models). Most of the time, __init_subclass__, class decorators, or descriptors are simpler alternatives.
