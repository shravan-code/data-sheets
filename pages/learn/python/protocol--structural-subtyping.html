Protocol — Structural Subtyping

Protocol (Duck Typing Formalized)

Protocol defines an interface without requiring inheritance. If it has the right methods, it satisfies the protocol.

from typing import Protocol, runtime_checkable

@runtime_checkable
class Drawable(Protocol):
    def draw(self) -> str: ...
    def get_area(self) -> float: ...

class Circle:
    def __init__(self, r: float): self.r = r
    def draw(self) -> str: return f'Circle(r={self.r})'
    def get_area(self) -> float: return 3.14159 * self.r ** 2

class Square:
    def __init__(self, s: float): self.s = s
    def draw(self) -> str: return f'Square(s={self.s})'
    def get_area(self) -> float: return self.s ** 2

# Both satisfy Drawable without inheriting from it!
def render(shape: Drawable) -> None:
    print(f'{shape.draw()}, area={shape.get_area():.2f}')

render(Circle(5))
render(Square(4))

# runtime_checkable enables isinstance checks
print(isinstance(Circle(1), Drawable))   # True
print(isinstance('hello', Drawable))     # False
Output
Circle(r=5), area=78.54
Square(s=4), area=16.00
True
