Descriptors

Descriptor Protocol

Descriptors implement __get__, __set__, __delete__ to customize attribute access. They power @property, @classmethod, @staticmethod.

# Non-data descriptor (only __get__)
class LoggedMethod:
    def __set_name__(self, owner, name):
        self.name = name
    
    def __get__(self, obj, objtype=None):
        if obj is None:
            return self
        print(f'Accessing {self.name}')
        return getattr(obj, f'_{self.name}')

# Data descriptor (has __get__ and __set__)
class ValidatedField:
    def __set_name__(self, owner, name):
        self.public_name = name
        self.private_name = f'_{name}'
    
    def __get__(self, obj, objtype=None):
        if obj is None:
            return self
        return getattr(obj, self.private_name, None)
    
    def __set__(self, obj, value):
        self.validate(value)
        setattr(obj, self.private_name, value)
    
    def validate(self, value):
        pass  # override in subclass

class PositiveInt(ValidatedField):
    def validate(self, value):
        if not isinstance(value, int) or value <= 0:
            raise ValueError(f'{self.public_name} must be positive int')

class Product:
    price = PositiveInt()
    quantity = PositiveInt()
    
    def __init__(self, price, qty):
        self.price = price
        self.quantity = qty

p = Product(10, 5)
print(p.price, p.quantity)

try:
    p.price = -1
except ValueError as e:
    print(e)

# How @property works (simplified)
class Temperature:
    def __init__(self, celsius):
        self._celsius = celsius
    
    @property
    def fahrenheit(self):
        return self._celsius * 9/5 + 32
    
    @fahrenheit.setter
    def fahrenheit(self, value):
        self._celsius = (value - 32) * 5/9

t = Temperature(100)
print(t.fahrenheit)     # 212.0
t.fahrenheit = 32
print(t._celsius)       # 0.0

Q:

What is the difference between a data descriptor and non-data descriptor?

A:

Data descriptor implements both __get__ and __set__ (or __delete__). Non-data descriptor implements only __get__. Data descriptors take priority over instance __dict__, so they always intercept access. Non-data descriptors are overridden by instance __dict__ entries. @property is a data descriptor; functions are non-data descriptors.
