total_ordering and cmp_to_key

total_ordering

Fill in missing comparison methods automatically.

from functools import total_ordering

@total_ordering
class Version:
    def __init__(self, major, minor, patch):
        self.major = major
        self.minor = minor
        self.patch = patch
    
    def __eq__(self, other):
        return (self.major, self.minor, self.patch) == \
               (other.major, other.minor, other.patch)
    
    def __lt__(self, other):  # only need eq + one comparison
        return (self.major, self.minor, self.patch) < \
               (other.major, other.minor, other.patch)
    
    def __repr__(self):
        return f'v{self.major}.{self.minor}.{self.patch}'

v1 = Version(1, 0, 0)
v2 = Version(2, 0, 0)
v3 = Version(1, 5, 3)

print(v1 < v2)    # True
print(v1 > v2)    # False  (generated!)
print(v1 <= v3)   # True   (generated!)
print(sorted([v2, v1, v3]))
