Basic Type Annotations

Variable and Function Annotations

Type hints make code self-documenting and enable static analysis tools like mypy.

from typing import Optional, Union, Any

# Variable annotations
name: str = 'Alice'
age: int = 30
height: float = 5.7
flags: list[bool] = [True, False]

# Function annotations
def greet(name: str, times: int = 1) -> str:
    return (name + ' ') * times

print(greet('Hello', 3))

# Optional (can be None)
def find_user(user_id: int) -> Optional[str]:
    users = {1: 'Alice', 2: 'Bob'}
    return users.get(user_id)  # returns str or None

print(find_user(1))
print(find_user(99))

# Union (multiple types)
def double(x: Union[int, float]) -> Union[int, float]:
    return x * 2

# Python 3.10+ syntax (preferred)
def double2(x: int | float) -> int | float:
    return x * 2

print(double(3))
print(double(2.5))
Output
Hello Hello Hello 
Alice
None
6
5.0

Generic Collections

Annotate collections with their element types.

from typing import List, Dict, Tuple, Set, FrozenSet
# Python 3.9+: use built-in list[int] instead of List[int]

def process_scores(scores: list[int]) -> dict[str, float]:
    return {
        'min': min(scores),
        'max': max(scores),
        'avg': sum(scores) / len(scores)
    }

print(process_scores([85, 92, 78, 96, 88]))

# Nested generics
Matrix = list[list[float]]

def transpose(matrix: Matrix) -> Matrix:
    return [list(row) for row in zip(*matrix)]

m = [[1.0, 2.0], [3.0, 4.0]]
print(transpose(m))

# Tuple with specific types
def get_coordinates() -> tuple[float, float]:
    return (37.7749, -122.4194)

# Variable-length tuple
def get_all_scores() -> tuple[int, ...]:
    return (85, 92, 78, 96)

coords = get_coordinates()
print(f'Lat: {coords[0]}, Lon: {coords[1]}')
Output
{'min': 78, 'max': 96, 'avg': 87.8}
[[1.0, 3.0], [2.0, 4.0]]
Lat: 37.7749, Lon: -122.4194
