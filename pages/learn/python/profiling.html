Profiling

timeit, cProfile, line_profiler

Measure where time is spent before optimizing.

import timeit
import cProfile
import pstats
import io

# timeit - micro-benchmarks
def list_append():
    lst = []
    for i in range(1000):
        lst.append(i)
    return lst

def list_comp():
    return [i for i in range(1000)]

# compare approaches
t1 = timeit.timeit(list_append, number=10000)
t2 = timeit.timeit(list_comp, number=10000)
print(f'append: {t1:.3f}s')
print(f'comp:   {t2:.3f}s')
print(f'comp is {t1/t2:.1f}x faster')

# timeit one-liner
result = timeit.timeit('[x**2 for x in range(100)]', number=100000)
print(f'{result:.3f}s')

# cProfile - full program profiling
def slow_function():
    total = 0
    for i in range(100000):
        total += i ** 2
    return total

pr = cProfile.Profile()
pr.enable()
result = slow_function()
pr.disable()

s = io.StringIO()
ps = pstats.Stats(pr, stream=s).sort_stats('cumulative')
ps.print_stats(5)
print(s.getvalue())
