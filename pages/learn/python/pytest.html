pytest

pytest Basics, Fixtures, Parametrize

pytest is more Pythonic — plain functions, powerful fixtures, and better output.

# test_calculator.py
import pytest

# Code under test
def divide(a, b):
    if b == 0:
        raise ZeroDivisionError('Cannot divide by zero')
    return a / b

# Simple test functions (no class needed)
def test_divide_basic():
    assert divide(10, 2) == 5.0

def test_divide_float():
    assert divide(7, 2) == pytest.approx(3.5)

def test_divide_by_zero():
    with pytest.raises(ZeroDivisionError, match='Cannot divide'):
        divide(5, 0)

# Fixtures - dependency injection for tests
@pytest.fixture
def sample_data():
    return [1, 2, 3, 4, 5]

@pytest.fixture(scope='module')  # shared across module
def db_connection():
    conn = {'connected': True}
    yield conn  # setup
    conn['connected'] = False  # teardown after all tests

def test_sum(sample_data):
    assert sum(sample_data) == 15

def test_max(sample_data):
    assert max(sample_data) == 5

# Parametrize - run test with multiple inputs
@pytest.mark.parametrize('a, b, expected', [
    (2, 3, 5),
    (0, 0, 0),
    (-1, 1, 0),
    (100, -50, 50),
])
def test_add(a, b, expected):
    assert a + b == expected

# Skip and xfail marks
@pytest.mark.skip(reason='Not implemented')
def test_future_feature():
    pass

@pytest.mark.xfail(reason='Known bug')
def test_known_bug():
    assert 1 == 2

# Running: pytest test_calculator.py -v
