File & Data Handling

CSV, JSON, pickle

Reading and writing structured data formats.

import csv, json, pickle
from io import StringIO

# CSV reading and writing
data = [
    {'name': 'Alice', 'dept': 'Engineering', 'salary': 90000},
    {'name': 'Bob',   'dept': 'Marketing',   'salary': 75000},
]

# Write CSV
with open('/tmp/employees.csv', 'w', newline='') as f:
    writer = csv.DictWriter(f, fieldnames=['name','dept','salary'])
    writer.writeheader()
    writer.writerows(data)

# Read CSV
with open('/tmp/employees.csv') as f:
    reader = csv.DictReader(f)
    for row in reader:
        print(dict(row))

# JSON - serialize/deserialize
config = {
    'host': 'localhost',
    'port': 5432,
    'options': ['ssl', 'timeout'],
    'credentials': {'user': 'admin'}
}

# Serialize
json_str = json.dumps(config, indent=2)
print(json_str[:80])

# Deserialize
parsed = json.loads(json_str)
print(parsed['options'])

# Custom encoder for non-serializable types
from datetime import datetime

class DateTimeEncoder(json.JSONEncoder):
    def default(self, obj):
        if isinstance(obj, datetime):
            return obj.isoformat()
        return super().default(obj)

data_with_date = {'created': datetime.now(), 'value': 42}
print(json.dumps(data_with_date, cls=DateTimeEncoder))
