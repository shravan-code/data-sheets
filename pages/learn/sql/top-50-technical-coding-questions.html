Section 30: Top 50 Technical Coding Questions

Top 50 Technical Coding Questions

All questions use common tables: employees(emp_id, name, dept_id, salary, manager_id, hire_date), departments(dept_id, dept_name), orders(order_id, customer_id, order_date, amount, status), customers(customer_id, name, city), products(product_id, name, category, price), order_items(order_id, product_id, quantity, price)

1 Find the second highest salary in the employees table.

SELECT MAX(salary) AS second_highest
FROM employees
WHERE salary < (SELECT MAX(salary) FROM employees);
Alternative Approach
SELECT salary FROM (
  SELECT salary, DENSE_RANK() OVER (ORDER BY salary DESC) AS rnk
  FROM employees
) t WHERE rnk = 2 LIMIT 1;

2 Find the Nth highest salary (generalized solution).

-- N = 3 (3rd highest)
SELECT salary FROM (
  SELECT salary, DENSE_RANK() OVER (ORDER BY salary DESC) AS rnk
  FROM employees
) t WHERE rnk = 3;

3 Find employees earning more than the average salary of their department.

SELECT e.name, e.salary, d.dept_name
FROM employees e
JOIN (
  SELECT dept_id, AVG(salary) AS avg_sal
  FROM employees
  GROUP BY dept_id
) dept_avg ON e.dept_id = dept_avg.dept_id
JOIN departments d ON e.dept_id = d.dept_id
WHERE e.salary > dept_avg.avg_sal;

4 Get the top 3 highest-paid employees per department.

SELECT emp_id, name, dept_id, salary FROM (
  SELECT emp_id, name, dept_id, salary,
    DENSE_RANK() OVER (PARTITION BY dept_id ORDER BY salary DESC) AS rnk
  FROM employees
) t WHERE rnk <= 3;

5 Find duplicate rows in the employees table based on name and dept_id.

SELECT name, dept_id, COUNT(*) AS cnt
FROM employees
GROUP BY name, dept_id
HAVING COUNT(*) > 1;

6 Delete duplicate rows keeping only the one with the lowest emp_id.

DELETE FROM employees
WHERE emp_id NOT IN (
  SELECT MIN(emp_id)
  FROM employees
  GROUP BY name, dept_id
);

-- PostgreSQL alternative using CTE:
WITH cte AS (
  SELECT emp_id, ROW_NUMBER() OVER (PARTITION BY name, dept_id ORDER BY emp_id) AS rn
  FROM employees
)
DELETE FROM employees WHERE emp_id IN (SELECT emp_id FROM cte WHERE rn > 1);

7 Find all employees who have no manager (top-level).

SELECT emp_id, name FROM employees WHERE manager_id IS NULL;

8 Find all managers (employees who manage at least one person).

SELECT DISTINCT m.emp_id, m.name
FROM employees e
JOIN employees m ON e.manager_id = m.emp_id;

9 Find departments with more than 5 employees.

SELECT d.dept_name, COUNT(e.emp_id) AS headcount
FROM departments d
JOIN employees e ON d.dept_id = e.dept_id
GROUP BY d.dept_name
HAVING COUNT(e.emp_id) > 5
ORDER BY headcount DESC;

10 Calculate running total of salary ordered by hire_date.

SELECT name, hire_date, salary,
  SUM(salary) OVER (ORDER BY hire_date ROWS BETWEEN UNBOUNDED PRECEDING AND CURRENT ROW) AS running_total
FROM employees
ORDER BY hire_date;

11 Find month-over-month revenue growth.

WITH monthly AS (
  SELECT DATE_TRUNC('month', order_date)::DATE AS month,
         SUM(amount) AS revenue
  FROM orders
  GROUP BY 1
)
SELECT month, revenue,
  LAG(revenue) OVER (ORDER BY month) AS prev_month,
  ROUND((revenue - LAG(revenue) OVER (ORDER BY month)) * 100.0 /
    NULLIF(LAG(revenue) OVER (ORDER BY month), 0), 2) AS growth_pct
FROM monthly
ORDER BY month;

12 Find customers who made orders every month for the last 3 months.

SELECT customer_id
FROM orders
WHERE order_date >= DATE_TRUNC('month', NOW()) - INTERVAL '2 months'
GROUP BY customer_id
HAVING COUNT(DISTINCT DATE_TRUNC('month', order_date)) = 3;

13 Find the gap in a sequence of order IDs (find missing IDs).

SELECT s.id + 1 AS missing_from,
       MIN(t.id) - 1 AS missing_to
FROM orders s
JOIN orders t ON s.id < t.id
GROUP BY s.id
HAVING s.id + 1 < MIN(t.id);

14 Find consecutive login days (islands of activity).

-- Step 1: Assign group by subtracting row number from date
WITH login_groups AS (
  SELECT user_id, login_date,
    login_date - (ROW_NUMBER() OVER (PARTITION BY user_id ORDER BY login_date))::INT AS grp
  FROM user_logins
),
consecutive AS (
  SELECT user_id, MIN(login_date) AS start_date, MAX(login_date) AS end_date,
    COUNT(*) AS consecutive_days
  FROM login_groups
  GROUP BY user_id, grp
)
SELECT * FROM consecutive WHERE consecutive_days >= 3;

15 Calculate the median salary.

-- PostgreSQL:
SELECT PERCENTILE_CONT(0.5) WITHIN GROUP (ORDER BY salary) AS median_salary
FROM employees;

-- Manual approach:
SELECT AVG(salary) AS median FROM (
  SELECT salary,
    COUNT(*) OVER () AS total,
    ROW_NUMBER() OVER (ORDER BY salary) AS rn
  FROM employees
) t
WHERE rn IN (FLOOR((total+1)/2.0), CEIL((total+1)/2.0));

16 Find products that have never been ordered.

SELECT p.product_id, p.name
FROM products p
LEFT JOIN order_items oi ON p.product_id = oi.product_id
WHERE oi.product_id IS NULL;

-- Alternative:
SELECT product_id, name FROM products
WHERE product_id NOT IN (SELECT DISTINCT product_id FROM order_items);

17 Find customers who bought product A but not product B.

SELECT DISTINCT customer_id FROM orders o
JOIN order_items oi ON o.order_id = oi.order_id
WHERE oi.product_id = 'A'
AND customer_id NOT IN (
  SELECT DISTINCT o2.customer_id FROM orders o2
  JOIN order_items oi2 ON o2.order_id = oi2.order_id
  WHERE oi2.product_id = 'B'
);

18 Write a query to transpose/pivot monthly sales into columns.

SELECT
  EXTRACT(YEAR FROM order_date) AS year,
  SUM(CASE WHEN EXTRACT(MONTH FROM order_date) = 1  THEN amount ELSE 0 END) AS Jan,
  SUM(CASE WHEN EXTRACT(MONTH FROM order_date) = 2  THEN amount ELSE 0 END) AS Feb,
  SUM(CASE WHEN EXTRACT(MONTH FROM order_date) = 3  THEN amount ELSE 0 END) AS Mar,
  SUM(CASE WHEN EXTRACT(MONTH FROM order_date) = 12 THEN amount ELSE 0 END) AS Dec
FROM orders
GROUP BY EXTRACT(YEAR FROM order_date)
ORDER BY year;

19 Find the department with the highest average salary.

SELECT d.dept_name, AVG(e.salary) AS avg_salary
FROM employees e
JOIN departments d ON e.dept_id = d.dept_id
GROUP BY d.dept_name
ORDER BY avg_salary DESC
LIMIT 1;

20 Compute a 3-day moving average of daily sales.

SELECT order_date,
  SUM(amount) AS daily_sales,
  AVG(SUM(amount)) OVER (
    ORDER BY order_date
    ROWS BETWEEN 2 PRECEDING AND CURRENT ROW
  ) AS moving_avg_3d
FROM orders
GROUP BY order_date
ORDER BY order_date;

21 Find employees hired in the last 30 days.

SELECT emp_id, name, hire_date
FROM employees
WHERE hire_date >= CURRENT_DATE - INTERVAL '30 days';

22 Calculate cumulative percentage of sales by product.

SELECT product_id,
  SUM(amount) AS total,
  ROUND(100.0 * SUM(amount) / SUM(SUM(amount)) OVER (), 2) AS pct,
  ROUND(100.0 * SUM(SUM(amount)) OVER (ORDER BY SUM(amount) DESC
    ROWS UNBOUNDED PRECEDING) / SUM(SUM(amount)) OVER (), 2) AS cumulative_pct
FROM order_items
GROUP BY product_id
ORDER BY total DESC;

23 Find all direct and indirect reports for a given manager (org hierarchy).

WITH RECURSIVE reports AS (
  SELECT emp_id, name, manager_id, 0 AS depth
  FROM employees
  WHERE emp_id = 1  -- starting manager

  UNION ALL

  SELECT e.emp_id, e.name, e.manager_id, r.depth + 1
  FROM employees e
  JOIN reports r ON e.manager_id = r.emp_id
)
SELECT * FROM reports WHERE emp_id != 1 ORDER BY depth, name;

24 Find the first purchase date for each customer.

SELECT customer_id,
  MIN(order_date) AS first_purchase_date
FROM orders
GROUP BY customer_id;

25 Find customers who have not placed an order in the last 90 days.

SELECT c.customer_id, c.name
FROM customers c
WHERE c.customer_id NOT IN (
  SELECT DISTINCT customer_id FROM orders
  WHERE order_date >= CURRENT_DATE - INTERVAL '90 days'
);

26 Rank customers by total spend within each city.

SELECT c.customer_id, c.name, c.city, SUM(o.amount) AS total_spend,
  RANK() OVER (PARTITION BY c.city ORDER BY SUM(o.amount) DESC) AS city_rank
FROM customers c
JOIN orders o ON c.customer_id = o.customer_id
GROUP BY c.customer_id, c.name, c.city;

27 Find pairs of employees in the same department.

SELECT a.name AS emp1, b.name AS emp2, a.dept_id
FROM employees a
JOIN employees b ON a.dept_id = b.dept_id AND a.emp_id < b.emp_id
ORDER BY a.dept_id, a.name;

28 Calculate the percentage of orders by status.

SELECT status,
  COUNT(*) AS cnt,
  ROUND(100.0 * COUNT(*) / SUM(COUNT(*)) OVER (), 2) AS pct
FROM orders
GROUP BY status
ORDER BY cnt DESC;

29 Find the most recent order for each customer.

SELECT customer_id, order_id, order_date, amount
FROM (
  SELECT *, ROW_NUMBER() OVER (PARTITION BY customer_id ORDER BY order_date DESC) AS rn
  FROM orders
) t
WHERE rn = 1;

30 Write a query to find the employee with the maximum salary in each department.

SELECT emp_id, name, dept_id, salary
FROM (
  SELECT *, RANK() OVER (PARTITION BY dept_id ORDER BY salary DESC) AS rnk
  FROM employees
) t WHERE rnk = 1;

31 Find all orders where the order amount is above the customer's average.

SELECT o.order_id, o.customer_id, o.amount,
  AVG(o.amount) OVER (PARTITION BY o.customer_id) AS cust_avg
FROM orders o
WHERE o.amount > (
  SELECT AVG(amount) FROM orders WHERE customer_id = o.customer_id
);

32 Get employee salary difference from their manager's salary.

SELECT e.name AS employee, e.salary AS emp_salary,
  m.name AS manager, m.salary AS mgr_salary,
  m.salary - e.salary AS salary_diff
FROM employees e
JOIN employees m ON e.manager_id = m.emp_id;

33 Find the category with the highest total revenue.

SELECT p.category, SUM(oi.quantity * oi.price) AS revenue
FROM order_items oi
JOIN products p ON oi.product_id = p.product_id
GROUP BY p.category
ORDER BY revenue DESC
LIMIT 1;

34 Write a query to detect salary anomalies (employees with salary > 2x dept average).

WITH dept_stats AS (
  SELECT dept_id, AVG(salary) AS avg_sal
  FROM employees GROUP BY dept_id
)
SELECT e.name, e.salary, d.avg_sal,
  ROUND(e.salary / d.avg_sal, 2) AS ratio
FROM employees e
JOIN dept_stats d ON e.dept_id = d.dept_id
WHERE e.salary > 2 * d.avg_sal;

35 Find customers who placed orders on consecutive days.

WITH daily AS (
  SELECT customer_id, order_date,
    LEAD(order_date) OVER (PARTITION BY customer_id ORDER BY order_date) AS next_date
  FROM (
    SELECT DISTINCT customer_id, order_date::DATE FROM orders
  ) t
)
SELECT DISTINCT customer_id
FROM daily
WHERE next_date = order_date + 1;

36 Write a query to compute year-over-year sales comparison.

WITH yearly AS (
  SELECT EXTRACT(YEAR FROM order_date)::INT AS yr, SUM(amount) AS revenue
  FROM orders GROUP BY 1
)
SELECT yr, revenue,
  LAG(revenue) OVER (ORDER BY yr) AS prev_year,
  ROUND((revenue - LAG(revenue) OVER (ORDER BY yr)) * 100.0 /
    NULLIF(LAG(revenue) OVER (ORDER BY yr), 0), 2) AS yoy_growth_pct
FROM yearly ORDER BY yr;

37 Implement SCD Type 2 upsert logic in SQL.

-- Expire old record:
UPDATE dim_customer
SET end_date = CURRENT_DATE - 1, is_current = FALSE
WHERE customer_id = 1 AND is_current = TRUE
  AND city != 'NYC';  -- only if value changed

-- Insert new record:
INSERT INTO dim_customer (customer_id, name, city, start_date, end_date, is_current)
SELECT 1, 'Alice', 'NYC', CURRENT_DATE, '9999-12-31', TRUE
WHERE NOT EXISTS (
  SELECT 1 FROM dim_customer WHERE customer_id = 1 AND is_current = TRUE AND city = 'NYC'
);

38 Find employees with the same salary.

SELECT salary, COUNT(*) AS cnt, STRING_AGG(name, ', ') AS employees
FROM employees
GROUP BY salary
HAVING COUNT(*) > 1
ORDER BY salary;

39 Find the total sales per month and the cumulative total.

SELECT
  DATE_TRUNC('month', order_date)::DATE AS month,
  SUM(amount) AS monthly_sales,
  SUM(SUM(amount)) OVER (ORDER BY DATE_TRUNC('month', order_date)) AS cumulative_sales
FROM orders
GROUP BY DATE_TRUNC('month', order_date)
ORDER BY month;

40 Get the top 5 customers by revenue and show what % of total revenue they represent.

WITH cust_rev AS (
  SELECT customer_id, SUM(amount) AS revenue FROM orders GROUP BY customer_id
),
total AS (
  SELECT SUM(amount) AS total_rev FROM orders
)
SELECT c.name, cr.revenue,
  ROUND(100.0 * cr.revenue / t.total_rev, 2) AS pct_of_total
FROM cust_rev cr
JOIN customers c ON cr.customer_id = c.customer_id
CROSS JOIN total t
ORDER BY cr.revenue DESC
LIMIT 5;

41 Write a query to get the longest streak of profitable days.

-- Using the gap-and-island technique:
WITH profitable AS (
  SELECT order_date::DATE AS day
  FROM orders
  GROUP BY order_date::DATE
  HAVING SUM(amount) > 0
),
with_grp AS (
  SELECT day,
    day - (ROW_NUMBER() OVER (ORDER BY day))::INT AS grp
  FROM profitable
)
SELECT grp, MIN(day) AS streak_start, MAX(day) AS streak_end,
  COUNT(*) AS streak_days
FROM with_grp
GROUP BY grp
ORDER BY streak_days DESC
LIMIT 1;

42 Write an incremental load query that only loads new/changed records.

-- Assume target has a last_updated_at tracking column
INSERT INTO target_orders (order_id, customer_id, amount, status, updated_at)
SELECT order_id, customer_id, amount, status, updated_at
FROM source_orders
WHERE updated_at > (
  SELECT COALESCE(MAX(updated_at), '1900-01-01') FROM target_orders
)
ON CONFLICT (order_id)
DO UPDATE SET
  amount = EXCLUDED.amount,
  status = EXCLUDED.status,
  updated_at = EXCLUDED.updated_at;

43 Find customers who have spent more than $10,000 lifetime and placed 5+ orders.

SELECT customer_id, COUNT(*) AS order_count, SUM(amount) AS lifetime_spend
FROM orders
GROUP BY customer_id
HAVING COUNT(*) >= 5 AND SUM(amount) > 10000
ORDER BY lifetime_spend DESC;

44 Write a query to create a date spine (every day for a year).

-- PostgreSQL:
SELECT generate_series(
  '2024-01-01'::DATE,
  '2024-12-31'::DATE,
  '1 day'::INTERVAL
)::DATE AS date_day;

-- Use as base to ensure no date gaps in reports:
WITH date_spine AS (
  SELECT generate_series('2024-01-01'::DATE, '2024-12-31'::DATE, '1 day') AS day
)
SELECT d.day, COALESCE(SUM(o.amount), 0) AS daily_revenue
FROM date_spine d
LEFT JOIN orders o ON o.order_date::DATE = d.day
GROUP BY d.day ORDER BY d.day;

45 Find the most purchased product in each category.

SELECT category, product_id, total_qty
FROM (
  SELECT p.category, oi.product_id, SUM(oi.quantity) AS total_qty,
    RANK() OVER (PARTITION BY p.category ORDER BY SUM(oi.quantity) DESC) AS rnk
  FROM order_items oi
  JOIN products p ON oi.product_id = p.product_id
  GROUP BY p.category, oi.product_id
) t WHERE rnk = 1;

46 Compute a cohort retention analysis (monthly cohorts).

WITH first_orders AS (
  SELECT customer_id, DATE_TRUNC('month', MIN(order_date)) AS cohort_month
  FROM orders GROUP BY customer_id
),
monthly_orders AS (
  SELECT DISTINCT o.customer_id,
    DATE_TRUNC('month', o.order_date) AS order_month,
    f.cohort_month,
    EXTRACT(EPOCH FROM DATE_TRUNC('month', o.order_date) - f.cohort_month) / (30*24*3600) AS months_since_join
  FROM orders o JOIN first_orders f ON o.customer_id = f.customer_id
)
SELECT cohort_month,
  months_since_join::INT AS month_number,
  COUNT(DISTINCT customer_id) AS active_customers
FROM monthly_orders
GROUP BY cohort_month, months_since_join
ORDER BY cohort_month, months_since_join;

47 Find employees who joined the same month as another employee (same dept).

SELECT a.name AS emp1, b.name AS emp2, a.dept_id,
  DATE_TRUNC('month', a.hire_date) AS hire_month
FROM employees a
JOIN employees b
  ON a.dept_id = b.dept_id
  AND DATE_TRUNC('month', a.hire_date) = DATE_TRUNC('month', b.hire_date)
  AND a.emp_id < b.emp_id;

48 Write a query to deduplicate a table and store results in a new table.

CREATE TABLE employees_deduped AS
SELECT DISTINCT ON (name, dept_id) *
FROM employees
ORDER BY name, dept_id, emp_id ASC;

-- Standard approach:
CREATE TABLE employees_deduped AS
SELECT * FROM (
  SELECT *, ROW_NUMBER() OVER (PARTITION BY name, dept_id ORDER BY emp_id) AS rn
  FROM employees
) t WHERE rn = 1;

49 Find the average time between orders for each customer.

WITH order_gaps AS (
  SELECT customer_id, order_date,
    LAG(order_date) OVER (PARTITION BY customer_id ORDER BY order_date) AS prev_date
  FROM orders
)
SELECT customer_id,
  AVG(order_date - prev_date) AS avg_days_between_orders
FROM order_gaps
WHERE prev_date IS NOT NULL
GROUP BY customer_id
ORDER BY avg_days_between_orders;

50 Write a full CDC-aware MERGE to sync a staging table to a target table.

-- PostgreSQL (using INSERT ... ON CONFLICT):
INSERT INTO target_employees (
  emp_id, name, dept_id, salary, updated_at, operation
)
SELECT
  s.emp_id, s.name, s.dept_id, s.salary, s.updated_at, s.operation
FROM staging_employees s
ON CONFLICT (emp_id) DO UPDATE SET
  name       = EXCLUDED.name,
  dept_id    = EXCLUDED.dept_id,
  salary     = EXCLUDED.salary,
  updated_at = EXCLUDED.updated_at,
  operation  = EXCLUDED.operation
WHERE target_employees.updated_at < EXCLUDED.updated_at;

-- Then handle deletes:
DELETE FROM target_employees
WHERE emp_id IN (
  SELECT emp_id FROM staging_employees WHERE operation = 'D'
);
