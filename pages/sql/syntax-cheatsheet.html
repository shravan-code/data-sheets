Section 28: Syntax Cheatsheet

Syntax Cheatsheet

Side-by-side syntax reference for PostgreSQL, MySQL, and MS SQL Server

Cheatsheet

topic postgresql mysql mssql
Auto Increment Primary Key emp_id SERIAL PRIMARY KEY -- or: emp_id INT GENERATED ALWAYS AS IDENTITY PRIMARY KEY emp_id INT AUTO_INCREMENT PRIMARY KEY emp_id INT IDENTITY(1,1) PRIMARY KEY
Current Date/Time SELECT CURRENT_DATE; -- date only SELECT NOW(); -- with time SELECT CURRENT_TIMESTAMP; -- same as NOW() SELECT CURDATE(); -- date only SELECT NOW(); -- with time SELECT SYSDATE(); SELECT CAST(GETDATE() AS DATE) -- date only SELECT GETDATE(); -- with time SELECT GETUTCDATE(); -- UTC
String Concatenation SELECT 'Hello' || ' ' || 'World'; SELECT CONCAT('Hello', ' ', 'World'); SELECT CONCAT('Hello', ' ', 'World'); -- || works only with PIPES_AS_CONCAT mode SELECT 'Hello' + ' ' + 'World'; SELECT CONCAT('Hello', ' ', 'World');
Limit Rows SELECT * FROM t ORDER BY id LIMIT 10 OFFSET 20; SELECT * FROM t ORDER BY id LIMIT 10 OFFSET 20; -- or: LIMIT 20, 10 SELECT TOP 10 * FROM t ORDER BY id; -- With offset: SELECT * FROM t ORDER BY id OFFSET 20 ROWS FETCH NEXT 10 ROWS ONLY;
Upsert / Merge INSERT INTO t (id, val) VALUES (1, 'a') ON CONFLICT (id) DO UPDATE SET val = EXCLUDED.val; INSERT INTO t (id, val) VALUES (1, 'a') ON DUPLICATE KEY UPDATE val = VALUES(val); -- or: REPLACE INTO t (id, val) VALUES (1, 'a'); MERGE INTO t AS target USING (VALUES (1, 'a')) AS src(id, val) ON target.id = src.id WHEN MATCHED THEN UPDATE SET val = src.val WHEN NOT MATCHED THEN INSERT (id, val) VALUES (src.id, src.val);
String to Date SELECT TO_DATE('15/01/2024', 'DD/MM/YYYY'); SELECT '2024-01-15'::DATE; SELECT STR_TO_DATE('15/01/2024', '%d/%m/%Y'); SELECT CAST('2024-01-15' AS DATE); SELECT CONVERT(DATE, '15/01/2024', 103); -- 103 = DD/MM/YYYY SELECT CAST('2024-01-15' AS DATE);
Date Arithmetic SELECT NOW() + INTERVAL '30 days'; SELECT NOW() - INTERVAL '1 year'; SELECT '2024-12-31'::DATE - '2024-01-01'::DATE; -- 365 SELECT DATE_ADD(NOW(), INTERVAL 30 DAY); SELECT DATE_SUB(NOW(), INTERVAL 1 YEAR); SELECT DATEDIFF('2024-12-31', '2024-01-01'); -- 365 SELECT DATEADD(DAY, 30, GETDATE()); SELECT DATEADD(YEAR, -1, GETDATE()); SELECT DATEDIFF(DAY, '2024-01-01', '2024-12-31'); -- 365
Conditional / NULL handling SELECT COALESCE(phone, 'N/A'); SELECT NULLIF(count, 0); SELECT CASE WHEN x > 0 THEN 'pos' ELSE 'neg' END; SELECT COALESCE(phone, 'N/A'); SELECT IFNULL(phone, 'N/A'); -- MySQL alias SELECT IF(x > 0, 'pos', 'neg'); SELECT COALESCE(phone, 'N/A'); SELECT ISNULL(phone, 'N/A'); -- MSSQL alias SELECT IIF(x > 0, 'pos', 'neg');
Safe Type Casting -- No built-in TRY_CAST; use regex or custom function: SELECT CASE WHEN col ~ '^[0-9]+$' THEN col::INT ELSE NULL END; -- No TRY_CAST; use CAST (throws on error): SELECT CAST(col AS UNSIGNED); SELECT TRY_CAST(col AS INT); -- NULL on failure SELECT TRY_CONVERT(INT, col); -- NULL on failure
String Aggregation SELECT dept_id, STRING_AGG(name, ', ' ORDER BY name) AS members FROM employees GROUP BY dept_id; SELECT dept_id, GROUP_CONCAT(name ORDER BY name SEPARATOR ', ') AS members FROM employees GROUP BY dept_id; SELECT dept_id, STRING_AGG(name, ', ') WITHIN GROUP (ORDER BY name) AS members FROM employees GROUP BY dept_id;
Window Functions SELECT name, salary, ROW_NUMBER() OVER (PARTITION BY dept_id ORDER BY salary DESC) AS rn FROM employees; -- Supported since MySQL 8.0: SELECT name, salary, ROW_NUMBER() OVER (PARTITION BY dept_id ORDER BY salary DESC) AS rn FROM employees; SELECT name, salary, ROW_NUMBER() OVER (PARTITION BY dept_id ORDER BY salary DESC) AS rn FROM employees;
Case-Insensitive Search SELECT * FROM t WHERE name ILIKE 'alice%'; -- ILIKE SELECT * FROM t WHERE LOWER(name) LIKE 'alice%'; -- Case-insensitive by default with utf8_general_ci: SELECT * FROM t WHERE name LIKE 'alice%'; SELECT * FROM t WHERE name COLLATE utf8_bin LIKE 'Alice%'; -- case-sensitive -- Case-insensitive by default with SQL_Latin1_General_CP1_CI_AS: SELECT * FROM t WHERE name LIKE 'alice%'; SELECT * FROM t WHERE name COLLATE Latin1_General_CS_AS LIKE 'Alice%'; -- case-sensitive
Create Index CREATE INDEX idx_name ON employees(name); CREATE UNIQUE INDEX idx_email ON employees(email); CREATE INDEX idx_partial ON employees(salary) WHERE status = 'active'; CREATE INDEX CONCURRENTLY idx_big ON big_table(col); -- non-blocking CREATE INDEX idx_name ON employees(name); CREATE UNIQUE INDEX idx_email ON employees(email); ALTER TABLE employees ADD INDEX idx_salary (salary); CREATE INDEX idx_name ON employees(name); CREATE UNIQUE INDEX idx_email ON employees(email); CREATE INDEX idx_include ON employees(dept_id) INCLUDE (name, salary); -- covering
Explain Query Plan EXPLAIN SELECT * FROM employees WHERE dept_id = 10; EXPLAIN ANALYZE SELECT * FROM employees WHERE dept_id = 10; EXPLAIN (FORMAT JSON) SELECT * FROM employees; EXPLAIN SELECT * FROM employees WHERE dept_id = 10; EXPLAIN FORMAT=JSON SELECT * FROM employees WHERE dept_id = 10; SET SHOWPLAN_TEXT ON; SELECT * FROM employees WHERE dept_id = 10; -- Or: use 'Include Actual Execution Plan' in SSMS
Truncate Table TRUNCATE TABLE employees; TRUNCATE TABLE employees RESTART IDENTITY; -- reset sequences TRUNCATE TABLE employees CASCADE; -- cascade to children TRUNCATE TABLE employees; -- auto_increment also resets TRUNCATE TABLE employees; -- IDENTITY also resets -- Cannot truncate if FK references exist
Temporary Tables CREATE TEMP TABLE tmp AS SELECT * FROM employees WHERE salary > 70000; CREATE TEMPORARY TABLE tmp (id INT, name TEXT); CREATE TEMPORARY TABLE tmp AS SELECT * FROM employees WHERE salary > 70000; -- Dropped when session ends SELECT * INTO #tmp FROM employees WHERE salary > 70000; -- session temp SELECT * INTO ##tmp FROM employees; -- global temp (all sessions) CREATE TABLE #tmp (id INT, name NVARCHAR(100));
Top N Per Group SELECT * FROM ( SELECT *, ROW_NUMBER() OVER (PARTITION BY dept_id ORDER BY salary DESC) AS rn FROM employees ) t WHERE rn <= 3; -- MySQL 8.0+: SELECT * FROM ( SELECT *, ROW_NUMBER() OVER (PARTITION BY dept_id ORDER BY salary DESC) AS rn FROM employees ) t WHERE rn <= 3; SELECT * FROM ( SELECT *, ROW_NUMBER() OVER (PARTITION BY dept_id ORDER BY salary DESC) AS rn FROM employees ) t WHERE rn <= 3;
Stored Procedure CREATE OR REPLACE PROCEDURE proc_name(param1 INT) LANGUAGE plpgsql AS $$ BEGIN UPDATE employees SET salary = salary * 1.1 WHERE dept_id = param1; END; $$; CALL proc_name(10); DELIMITER // CREATE PROCEDURE proc_name(IN dept INT) BEGIN UPDATE employees SET salary = salary * 1.1 WHERE dept_id = dept; END // DELIMITER ; CALL proc_name(10); CREATE PROCEDURE proc_name @dept INT AS BEGIN UPDATE employees SET salary = salary * 1.1 WHERE dept_id = @dept; END; EXEC proc_name @dept = 10;
